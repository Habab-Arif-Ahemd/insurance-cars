import { Driver } from "./Driver";

export interface QuoteRequest {
  policyEffectiveDate?: string; // Next day
  nationalityId?: number;
  insuredIdentityTypeId?: number; // First digit from 'insuredIdentityNumber' (1 or 2)
  insuredIdentityNumber?: number; // ((landing form) National ID)
  insuredBirthDate?: string;
  insuredGenderId?: string; // (M or F)
  insuredSocialStatusId?: number; // (single=1, married=2)
  InsuredEducationLevelId?: number; 
  drivingCityId?: number;
  promoCode?: number;
  childrenUnder16Years?: number; // 1-10 (0 if single)

  vehicleIdTypeId?: number; // ((landing form) form=1, customcard=2)
  vehicleId?: number; // ((landing form) seq num)
  isVehicleOwnerTransfer?: boolean; // (extra step, transfer=true, new=false)
  vehicleSumInsured?: number;
  vehicleRepairMethodId?: number; // (Add to insurance steps)
  vehicleUseId?: number;
  insuredWorkCityId?: number;
  vehicleTransmissionTypeId?: number;
  vehicleMileageExpectedAnnualId?: number;
  vehicleParkingLocationId?: number;
  isVehicleModified?: boolean;
  vehicleModificationDetails?: string;
  vehicleCurrentMileage?: number;
  vehicleOwnerIdentityNumber?: number;
  vehicleSpecifications?: number[];
  drivers?: Driver[];
}
