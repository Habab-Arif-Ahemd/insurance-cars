import { Injectable } from "@angular/core";
import { BehaviorSubject } from "rxjs";
import { QuoteRequest } from "src/app/models/insurance-steps/QuoteRequest";
import { LocallyStoredItemsKeys } from "src/app/models/app/LocallyStoredItemsKeys";
import { InsuranceStepsData } from "src/app/models/master-table/InsuranceStepsData";
import { Driver } from "src/app/models/insurance-steps/Driver";

@Injectable({
  providedIn: "root"
})
export class InsuranceStepsService {
  totalInsuranceSteps = 7;

  currentStepNum: BehaviorSubject<number> = new BehaviorSubject<number>(0);
  userQuoteRequest: BehaviorSubject<QuoteRequest> = new BehaviorSubject<
    QuoteRequest
  >({} as QuoteRequest);
  insuranceStepsData: BehaviorSubject<InsuranceStepsData> = new BehaviorSubject<
    InsuranceStepsData
  >({} as InsuranceStepsData);
 
  FormsValidationStatus: BehaviorSubject<
    { formName: string; isValid: boolean }[]
  > = new BehaviorSubject<{ formName: string; isValid: boolean }[]>([]);

  constructor() {}

  /**
   * Determines current step number to calculate the completion percentage
   * @param stepNumber Current step #, incremented by 1
   */
  public setCurrentStepNum(stepNumber: number): void {
    this.currentStepNum.next(stepNumber);
    localStorage.setItem(
      LocallyStoredItemsKeys.InsuranceCurrentStepNumber,
      stepNumber.toString()
    );
  }

  /**
   * Returns the current step number behavior subject
   */
  public getCurrentStepNum(): BehaviorSubject<number> {
    // Get from local storage
    const storedStepNum = localStorage.getItem(
      LocallyStoredItemsKeys.InsuranceCurrentStepNumber
    );
    // If exists in local storage then set it's value on the subject
    if (storedStepNum && storedStepNum.length > 0) {
      this.setCurrentStepNum(parseInt(storedStepNum));
    }
    // Return the subject
    return this.currentStepNum;
  }

  /**
   * Set the insurance quote request data object for the final quote request
   * @param userQuoteRequest Current quote request data
   */
  public setUserQuoteRequest(userQuoteRequest: QuoteRequest): void {
    const currentQuoteReq = JSON.parse(
      localStorage.getItem(LocallyStoredItemsKeys.UserQuoteRequest)
    );
    if (currentQuoteReq) {
      Object.keys(currentQuoteReq).forEach(key => {
        if (!userQuoteRequest[key]) {
          userQuoteRequest[key] = currentQuoteReq[key];
        }
      });
    }
    this.userQuoteRequest.next(userQuoteRequest);
    localStorage.setItem(
      LocallyStoredItemsKeys.UserQuoteRequest,
      JSON.stringify(userQuoteRequest)
    );
  }

  public overridenSetQuoteRequest(userQuoteRequest: QuoteRequest): void {
    this.userQuoteRequest.next(userQuoteRequest);
    localStorage.setItem(
      LocallyStoredItemsKeys.UserQuoteRequest,
      JSON.stringify(userQuoteRequest)
    );
  }

  public setUserQuoteReqDrivers(drivers: Driver[]): void {
    let currentQuoteReq = this.getUserQuoteRequest().value;
    let currentDrivers: Driver[];

    if (currentQuoteReq.drivers) {
      currentDrivers = currentQuoteReq.drivers.concat();
      currentDrivers = currentDrivers.concat(drivers);
    } else {
      currentDrivers = drivers.concat();
    }

    currentQuoteReq.drivers = currentDrivers;
    this.userQuoteRequest.next(currentQuoteReq);
    localStorage.setItem(
      LocallyStoredItemsKeys.UserQuoteRequest,
      JSON.stringify(currentQuoteReq)
    );
  }

  /**
   * Returns the current quote request object data
   */
  public getUserQuoteRequest(): BehaviorSubject<QuoteRequest> {
    // Get from local storage
    const storedQuoteReqData = localStorage.getItem(
      LocallyStoredItemsKeys.UserQuoteRequest
    );
    // If exists in local storage then set it's value on the subject
    if (storedQuoteReqData && storedQuoteReqData.length > 0) {
      this.setUserQuoteRequest(JSON.parse(storedQuoteReqData));
    }
    return this.userQuoteRequest;
  }

  /**
   * Set the insurance steps data master tables data
   * @param insuranceStepsData Steps data
   */
  public setInsuranceStepsData(insuranceStepsData: InsuranceStepsData): void {
    this.insuranceStepsData.next(insuranceStepsData);
    localStorage.setItem(LocallyStoredItemsKeys.InsuranceCurrentStepData, JSON.stringify(insuranceStepsData));
  }

  /**
   * Returns the current insurance data
   */
  public getInsuranceStepsData(): BehaviorSubject<InsuranceStepsData> {
    // Get from local storage
    // const storedStepsData = localStorage.getItem(LocallyStoredItemsKeys.InsuranceCurrentStepData);
    // // If exists in local storage then set it's value on the subject
    // if (storedStepsData && storedStepsData.length > 0) {
    //    this.setInsuranceStepsData(JSON.parse(storedStepsData));
    // }
    return this.insuranceStepsData;
  }

  public addInsuranceStepsObject(res: any, field: string): void {
    // Get current steps
    const tempStepsData = this.getInsuranceStepsData().value;
    // Add the object
    tempStepsData[field] = res;
    // Set the steps
    this.setInsuranceStepsData(tempStepsData);
  }

 
  public setFormsValidate(
    formsValidation: { formName: string; isValid: boolean }[]
  ): void {
    this.FormsValidationStatus.next(formsValidation);
  }
 
}
