import { Injectable } from "@angular/core";
import { PreliminaryDataRequest } from "src/app/payload/requests/quotation/PreliminaryDataRequest";
import { InsuranceStepsService } from "./insurance-steps.service";
import { InsuranceStepsApiService } from "./insurance-steps-api.service";
import { AppRoutes } from "src/app/models/app/AppRoutes";
import { Router } from "@angular/router";
import { BehaviorSubject } from 'rxjs/internal/BehaviorSubject';

@Injectable({
  providedIn: "root"
})
export class InquireService {
  isLoading: BehaviorSubject<boolean> = new BehaviorSubject<boolean>(false);
  errorMessage: BehaviorSubject<string> = new BehaviorSubject<string>('');

  constructor(
    private insuranceStepsService: InsuranceStepsService,
    private insuranceStepsApi: InsuranceStepsApiService,
    private router: Router
  ) {}
  inquire(inquireDateRequest: PreliminaryDataRequest) {
    // Show loading spinner
    this.isLoading.next(true);
    // Construct request data which is used to retreive the preliminary quote order data
    const preliminaryDataRequest: PreliminaryDataRequest = inquireDateRequest;
    // Add the data to the quotation request construction object
    this.insuranceStepsService.setUserQuoteRequest(preliminaryDataRequest);
    // Attempt to get the user's previous quotation request preliminary data
    this.getUserPreliminaryData(preliminaryDataRequest);
  }

  private getUserPreliminaryData(
    preliminaryDataRequest: PreliminaryDataRequest
  ): void {
    // Send the request
    this.insuranceStepsApi
      .getUserQuotePreliminaryData(preliminaryDataRequest)
      .subscribe(
        preliminaryData => {
          // Add the prelimary data to the quotation request construction object
          this.insuranceStepsService.setUserQuoteRequest(preliminaryData);
          // Navigate the user
          this.navigateUser();

          // Hide spinner
          this.isLoading.next(false);
        },
        err => {
          // If preliminary data was not found
          if (err.error.statusCode === 404) {
            // Navigate the user anyway
            this.navigateUser();
          } else {
            // Show error message
            this.errorMessage = err.error.responseMessage;
          }

          // Hide spinner
          this.isLoading.next(false);
        }
      );
  }

  private navigateUser() {
    this.router.navigate([AppRoutes.insuranceSteps]);
    // Increase insurance steps counter
    this.insuranceStepsService.setCurrentStepNum(1);
  }
}
