import { NgModule } from '@angular/core';
import { CommonModule } from '@angular/common';
import { HttpClient } from '@angular/common/http';
import { FormsModule, ReactiveFormsModule } from '@angular/forms';
import { TranslateModule, TranslateLoader } from '@ngx-translate/core';
import { TranslateHttpLoader } from '@ngx-translate/http-loader';
import { NgbModule } from '@ng-bootstrap/ng-bootstrap';
import { LoadingComponent } from './pages/loading/loading.component';
import { AuthenticationPageComponent } from '../Auth/auth.component';
import { ErrorComponent } from './pages/error/error.component';
import { ProgressComponent } from './layout/progress/progress.component';
import { QuotesListComponent } from '../quotes/quotes-list/quotes-list.component';
import { QuoteComponent } from '../quotes/quote/quote.component';
import { NgSelectModule } from '@ng-select/ng-select';


export function HttpLoaderFactory(httpClient: HttpClient) {
   return new TranslateHttpLoader(httpClient);
}


@NgModule({
   declarations: [
      AuthenticationPageComponent,
      LoadingComponent,
      ErrorComponent,
      ProgressComponent,
      QuotesListComponent,
      QuoteComponent
   ],
   imports: [
      CommonModule,
      FormsModule,
      ReactiveFormsModule,
      NgbModule,
      NgSelectModule,
      TranslateModule.forRoot({
         loader: {
           provide: TranslateLoader,
           useFactory: HttpLoaderFactory,
           deps: [HttpClient]
         }
      })
   ],
   exports: [
      TranslateModule,
      NgbModule,
      FormsModule,
      AuthenticationPageComponent,
      LoadingComponent,
      ErrorComponent,
      QuotesListComponent,
      QuoteComponent
   ]
})


export class SharedModule { }
