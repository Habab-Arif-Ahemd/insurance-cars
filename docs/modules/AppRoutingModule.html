import { NgModule } from '@angular/core';
import { Routes, RouterModule } from '@angular/router';
import { AuthenticationPageComponent } from './components/Auth/auth.component';
import { LandingComponent } from './components/landing/landing.component';
import { QuotesPageComponent } from './components/quotes/quotes-page/quotes-page.component';
import { QuotationRequestPageComponent } from './components/quotation-request/quotation-request-page/quotation-request-page.component';
import { ProfileComponent } from './components/user/profile/profile.component';
import { PoliciesPanelComponent } from './components/user/policies/policies-panel/policies-panel.component';
import { VehiclesComponent } from './components/user/vehicles/vehicles.component';
import { BoughtPoliciesComponent } from './components/user/policies/bought-policies/bought-policies.component';
import { ExpiringPoliciesComponent } from './components/user/policies/expiring-policies/expiring-policies.component';
import { ExpiredPoliciesComponent } from './components/user/policies/expired-policies/expired-policies.component';
import { OrderReviewComponent } from './components/order-review/order-review.component';
import { AppRoutes } from './models/app/AppRoutes';
import { DirectAccessGuard } from './helpers/guards/direct-access.guard';
import { AuthGuard } from './helpers/guards/auth.guard';
import { ErrorComponent } from './components/shared/pages/error/error.component';
import { AccountPanelComponent } from './components/user/account/account-panel/account-panel.component';
import { PrevQuotesComponent } from './components/user/prev-quotes/prev-quotes.component';
import { VehiclesListComponent } from './components/user/vehicles-list/vehicles-list.component';
import { TechnicalSupportComponent } from './components/user/technical-support/technical-support.component';
import { PaymentSuccessComponent } from './components/payment-success/payment-success.component';


const routes: Routes = [
   { path: 'success-payment', component: PaymentSuccessComponent},
   { path: '', component: LandingComponent},
   { path: 'error', component: ErrorComponent},
   { path: 'account/login', component: AuthenticationPageComponent },
   { path: 'account/verify', component: AuthenticationPageComponent },
   { path: 'account/reset', component: AuthenticationPageComponent },
   { path: 'quotes', component: QuotesPageComponent/* , canActivate: [DirectAccessGuard] */},
   { path: 'quotation', component: QuotationRequestPageComponent },
   { path: 'checkout', component: OrderReviewComponent, /* canActivate: [DirectAccessGuard, AuthGuard] */ },
   { path: 'checkout/success', component: PaymentSuccessComponent },
   /* Control panel */
   { path: 'user', component: ProfileComponent , canActivate: [AuthGuard] , children: [
      /* Quotes */
      { path: 'previous-quotes', component: PrevQuotesComponent },
      /* Policies */
      { path: 'policies', component: PoliciesPanelComponent },
      { path: 'bought', component: BoughtPoliciesComponent },
      { path: 'expiring', component: ExpiringPoliciesComponent },
      { path: 'expired', component: ExpiredPoliciesComponent },
      { path: 'support', component: TechnicalSupportComponent },

      /* Vehicles */
      { path: 'vehicles', component: VehiclesListComponent },
      { path: 'new-quote/select-vehicle', component: VehiclesComponent },

      /* Account Settings */
      { path: 'account', component: AccountPanelComponent },
      { path: 'account' + '/:activePanel', component: AccountPanelComponent },
      { path: 'account' + '/:activePanel/:state', component: AccountPanelComponent, canActivate: [DirectAccessGuard] }
   ]},
];


@NgModule({
   imports: [RouterModule.forRoot(routes)],
   exports: [RouterModule]
})


export class AppRoutingModule { }
