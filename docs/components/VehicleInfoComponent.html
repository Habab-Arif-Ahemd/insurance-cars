import { ChangeDetectionStrategy, Component, EventEmitter, OnDestroy, OnInit, Output } from "@angular/core";
import { FormBuilder, FormGroup, Validators } from "@angular/forms";
import { Scroll } from "src/app/helpers/utils/scroll";
import { Driver } from "src/app/models/insurance-steps/Driver";
import { QuoteRequest } from "src/app/models/insurance-steps/QuoteRequest";
import { AppService } from "src/app/services/app.service";
import { InsuranceStepsService } from "src/app/services/insurance-steps/insurance-steps.service";
import { FormsNames } from 'src/app/models/insurance-steps/StepsFormsName';
import { FormsValidationService } from 'src/app/services/insurance-steps/forms-validation.service';
import AppUtils from 'src/app/helpers/utils/AppUtils';

@Component({
 // changeDetection: ChangeDetectionStrategy.Default,
  selector: "app-vehicle-info",
  templateUrl: "./vehicle-info.component.html",
  styleUrls: [
    "../quotation-request-page/quotation-request-page.component.css",
    "./vehicle-info.component.css",
  ],
})
export class VehicleInfoComponent implements OnInit, OnDestroy {
  
  @Output() scroll = new EventEmitter();

  vehicleInfoForm: FormGroup;
  vehicleUpdated = true;
  stepNumber;
  stepsData;
  submitted = false;
 
  canUpdate = false;
  formUpdate = false;
  lang: string;
  previousQuote: QuoteRequest;
  previousQuoteDriver: Driver;
  termSearchFn = AppUtils.searchItemTerm;

  constructor(
    private insuranceStepService: InsuranceStepsService,
    private fb: FormBuilder,
    private scrollOnErrer: Scroll,
    private appService: AppService,
    private formvalidationService: FormsValidationService
  ) {}

  ngOnInit() {
    this.appService.getAppLang().subscribe((language) => {
      this.lang = language;
    });
    this.getStepData();
    this.initForm();
    this.watchStepNum();

    this.watchQuotationRequestSubmit();
    this.getPreviousQuoteRequest();
    this.watchFormAfterSubmit();
  }

  initForm(): void {
    this.vehicleInfoForm = this.fb.group({
      vehicleSumInsured: [
        null,
        [Validators.required, Validators.pattern("^[0-9,]*$")],
      ],
      drivingCityId: [null, [Validators.required]],
      vehicleUseId: [null, [Validators.required]],
      vehicleTransmissionTypeId: [null, [Validators.required]],
      vehicleParkingLocationId: [null, [Validators.required]],
      isVehicleModified: [false],
      vehicleModificationDetails: [null],
      vehicleMileageExpectedAnnualId: [null, [Validators.required]],
      vehicleCurrentMileage:[
        null,
        [Validators.required, Validators.pattern("^[0-9,]*$")],
      ],
    
    });
  }
  // convenience getter for easy access to form fields
  get f() {
    return this.vehicleInfoForm.controls;
  }

  submitForm() {
    this.submitted = true;

    if (this.vehicleInfoForm.invalid) {
      const firstElementWithError = document.querySelector(
        ".ng-invalid[formControlName]"
      );
      this.scrollOnErrer.scrollTo(firstElementWithError);
      return;
    }
    // remove commas from numbers
    this.vehicleInfoForm.value.vehicleSumInsured=   parseInt(this.vehicleInfoForm.value.vehicleSumInsured.replace(/,/g, ""));
    this.vehicleInfoForm.value.vehicleCurrentMileage=  parseInt(this.vehicleInfoForm.value.vehicleCurrentMileage.replace(/,/g, "")); 
    
    if (!this.vehicleInfoForm.value.isVehicleModified) {
      this.vehicleInfoForm.value.vehicleModificationDetails = null;
    }
  
    this.insuranceStepService.setUserQuoteRequest(this.vehicleInfoForm.value);
    if (this.stepNumber == 2) { 
      this.insuranceStepService.setCurrentStepNum(3) 
      this.formvalidationService.addFormsStatus({
        formName: FormsNames.vehicleInfoForm,
        isValid: true
      });
    } 
  }

  getStepData() {
    this.insuranceStepService.getInsuranceStepsData().subscribe((stepsData) => {
      this.stepsData = stepsData;
    });
  }

  vehicleIsUpdated() {
    this.vehicleUpdated = !this.vehicleUpdated;
    this.vehicleInfoForm.get("vehicleModificationDetails").reset();
    if (!this.vehicleUpdated) {
      this.vehicleInfoForm.controls[
        "vehicleModificationDetails"
      ].setValidators([Validators.required]);
      this.vehicleInfoForm.controls[
        "vehicleModificationDetails"
      ].updateValueAndValidity();
    } else {
      this.vehicleInfoForm.controls[
        "vehicleModificationDetails"
      ].clearValidators();
      this.vehicleInfoForm.controls[
        "vehicleModificationDetails"
      ].updateValueAndValidity();
    }
  }

  watchStepNum() {
    this.insuranceStepService.getCurrentStepNum().subscribe((stepNum) => {
      this.stepNumber = stepNum;
    });
  }

  watchQuotationRequestSubmit() {
    /*  this.eventsSubscription = this.eventsSubmit.subscribe(() => {

      this.submitForm();
    }) */
  }

  getPreviousQuoteRequest() {
    this.previousQuote = this.insuranceStepService.getUserQuoteRequest().value;
    this.previousQuote.isVehicleModified
      ? this.vehicleIsUpdated()
      : (this.vehicleUpdated = true);
    if (this.previousQuote.drivers) {
      this.previousQuote.drivers.forEach((drivers) => {
        if (drivers.typeId == 1) {
          this.previousQuoteDriver = drivers;
        }
      });
    }
  }

  watchFormAfterSubmit() {
    this.vehicleInfoForm.valueChanges.subscribe(() => {
      if (this.vehicleInfoForm.valid && this.stepNumber >= 3) {
        this.formvalidationService.addFormsStatus({
          formName: FormsNames.vehicleInfoForm,
          isValid: true
        });
        this.submitForm();
      } else {
        this.formvalidationService.addFormsStatus({
          formName: FormsNames.vehicleInfoForm,
          isValid: false
        });
      }   
    })
}


  ngOnDestroy() {
    /* this.eventsSubscription.unsubscribe(); */
  }
}
