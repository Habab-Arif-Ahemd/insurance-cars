import {
  ChangeDetectionStrategy,
  Component,
  EventEmitter,
  OnDestroy,
  OnInit,
  Output,
} from "@angular/core";
import { FormBuilder, FormGroup, Validators } from "@angular/forms";
import { QuoteRequest } from "src/app/models/insurance-steps/QuoteRequest";
import { AppService } from "src/app/services/app.service";
import { InsuranceStepsService } from "src/app/services/insurance-steps/insurance-steps.service";
import { Scroll } from "../../../helpers/utils/scroll";
import { FormsValidationService } from 'src/app/services/insurance-steps/forms-validation.service';
import { FormsNames } from 'src/app/models/insurance-steps/StepsFormsName';
import AppUtils from 'src/app/helpers/utils/AppUtils';


@Component({
  selector: "app-insurance-info",
  templateUrl: "./insurance-info.component.html",
  styleUrls: [
    "../quotation-request-page/quotation-request-page.component.css",
    "./insurance-info.component.css",
  ],
})
export class InsuranceInfoComponent implements OnInit, OnDestroy {
  /*  @Input() eventsSubmit: Observable<void>;
  private eventsSubscription: Subscription; */

  @Output() scroll = new EventEmitter();

  insuranceInfoForm: FormGroup;
  submitted = false;
  stepNumber: number;
  ownership = false;
  stepsData = null;
  cuurentStepData;

  insuredIdentityNumber;

  isLoading = true;
  canUpdate = false;
  formUpdate = false;
  now = new Date();
  minDate;
  maxDate;
  model;
  lang;
  vehicleRepairMethods: any;
  previousQuote: QuoteRequest;
  previousQuoteEffectiveDate: Date;

  termSearchFn = AppUtils.searchItemTerm;

  constructor(
    private insuranceSteps: InsuranceStepsService,
    private fb: FormBuilder,
    private scrollOnErrer: Scroll,
    private appService: AppService,
    private formvalidationService: FormsValidationService
  ) {}

  ngOnInit() {
    this.insuranceSteps.setCurrentStepNum(1);
    this.appService.getAppLang().subscribe((language) => {
      this.lang = language;  
    });

    this.getStepData();
    this.initForm();
    this.setMinDate();
    this.setMaxDate();
    this.watchStepNum();
    this.getInsuredIdentityNumber();
    this.getPreviousQuoteRequest();
    // this.validateIdentityNo();
    this.watchFormAfterSubmit();
  }

  // convenience getter for easy access to form fields
  get f() {
    return this.insuranceInfoForm.controls;
  }

  initForm(): void {
    this.insuranceInfoForm = this.fb.group({
      isVehicleOwnerTransfer: [false],
      policyEffectiveDate: [null, [Validators.required]],
      vehicleRepairMethodId: [null, [Validators.required]],
      vehicleOwnerIdentityNumber: [null],
      insuredIdentityNumber: [this.insuredIdentityNumber],
      vehicleEngineSizeId: [null],
      nationalityId: [null],
      promoCode: [null],
    });
  }

  getStepData() {
    this.insuranceSteps.getInsuranceStepsData().subscribe((stepsData) => {
      /*  console.log(stepsData) */
      this.isLoading = false;
      this.stepsData = stepsData;
    });
  }

  getInsuredIdentityNumber() {
    this.insuranceSteps.getUserQuoteRequest().subscribe((stepsData) => {
      this.insuredIdentityNumber = stepsData.insuredIdentityNumber;
    });
  }

  getPreviousQuoteRequest() {
    this.previousQuote = this.insuranceSteps.getUserQuoteRequest().value;
    this.previousQuote.isVehicleOwnerTransfer
      ? this.isTransfer()
      : (this.ownership = false);
  }

  setMinDate() {
    let minDate = AppUtils.addDays(this.now, 1);
    this.minDate = {
      year: minDate.getFullYear(),
      month: minDate.getMonth() + 1,
      day: minDate.getDate(),
    };
  }

  setMaxDate() {
    let maxDate = AppUtils.addDays(this.now, 14);
    this.maxDate = {
      year: maxDate.getFullYear(),
      month: maxDate.getMonth() + 1,
      day: maxDate.getDate(),
    };
  }

  submitForm() {
    console.log(this.insuranceInfoForm);

    this.submitted = true;
    if (this.insuranceInfoForm.invalid) {
      const firstElementWithError = document.querySelector(".ng-invalid");
      this.scrollOnErrer.scrollTo(firstElementWithError);
      return;
    }

    if (this.insuranceInfoForm.value.vehicleOwnerIdentityNumber == null) {
      this.insuranceInfoForm.value.vehicleOwnerIdentityNumber = this.insuredIdentityNumber;
    } else {
      this.insuranceInfoForm.value.vehicleOwnerIdentityNumber = this.insuranceInfoForm.value.vehicleOwnerIdentityNumber;
    }
    let policyEffectiveDate = this.insuranceInfoForm.value.policyEffectiveDate;
    this.insuranceInfoForm.value.policyEffectiveDate = new Date(
      policyEffectiveDate.year,
      policyEffectiveDate.month - 1,
      policyEffectiveDate.day + 1  
    ).toISOString();
    this.insuranceSteps.setUserQuoteRequest(this.insuranceInfoForm.value);
    if (this.stepNumber == 1) {
      this.insuranceSteps.setCurrentStepNum(2);
      this.formvalidationService.addFormsStatus({
        formName: FormsNames.insuranceInfoForm,
        isValid: true
      });
    }  
    this.submitted = false;
  }

  isTransfer() {
    this.insuranceInfoForm.get("vehicleOwnerIdentityNumber").reset();
    this.ownership = !this.ownership;
    this.submitted = false;

    if (this.ownership) {
      this.insuranceInfoForm
        .get("vehicleOwnerIdentityNumber")
        .setValidators(
          Validators.compose([Validators.required, Validators.minLength(10)])
        );

      this.insuranceInfoForm
        .get("vehicleOwnerIdentityNumber")
        .updateValueAndValidity();
    } else {
      this.insuranceInfoForm.controls[
        "vehicleOwnerIdentityNumber"
      ].clearValidators();
    }
    this.insuranceInfoForm
      .get("vehicleOwnerIdentityNumber")
      .updateValueAndValidity();
  }

  watchStepNum() {
    this.insuranceSteps.getCurrentStepNum().subscribe((stepNum) => {
      this.stepNumber = stepNum;
    });
  }

 

 /*  validateIdentityNo() {
    if (this.previousQuote.drivers) {
      const previousQuoteIdentityNo: number[] = this.previousQuote.drivers.map(
        (id) => id.identityNumber
      );
      previousQuoteIdentityNo.forEach((id) => {
        this.driverService.addDriverIdentityNo({
          identityNumber: id,
        });
      });
    }
  } */

    watchFormAfterSubmit() {
      this.insuranceInfoForm.valueChanges.subscribe(() => {
        if (this.insuranceInfoForm.valid && this.stepNumber >= 2) {
          this.formvalidationService.addFormsStatus({
            formName: FormsNames.insuranceInfoForm,
            isValid: true
          });
          this.submitForm();
        } else {
          this.formvalidationService.addFormsStatus({
            formName: FormsNames.insuranceInfoForm,
            isValid: false
          });
        }
       
         
        
      })
  }

  ngOnDestroy() {
    /*  this.eventsSubscription.unsubscribe(); */
  }
}
