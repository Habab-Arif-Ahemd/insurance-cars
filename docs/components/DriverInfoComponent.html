import {
  ChangeDetectionStrategy,
  Component,
  EventEmitter,
  OnInit,
  Output,
  ViewChild,
  ElementRef,
} from "@angular/core";
import { FormBuilder, FormGroup, Validators } from "@angular/forms";
import DropDownValues from "src/app/helpers/utils/DropDownValues";
import { Scroll } from "src/app/helpers/utils/scroll";
import { Driver } from "src/app/models/insurance-steps/Driver";
import { QuoteRequest } from "src/app/models/insurance-steps/QuoteRequest";
import { AppService } from "src/app/services/app.service";
import { InsuranceStepsService } from "src/app/services/insurance-steps/insurance-steps.service";
import { FormsNames } from "src/app/models/insurance-steps/StepsFormsName";
import { FormsValidationService } from "src/app/services/insurance-steps/forms-validation.service";
import { NgbActiveModal, NgbModal } from '@ng-bootstrap/ng-bootstrap';
import AppUtils from 'src/app/helpers/utils/AppUtils';


@Component({
  selector: "app-driver-info",
  templateUrl: "./driver-info.component.html",
  styleUrls: [
    "./driver-info.component.css",
    "../../quotation-request-page/quotation-request-page.component.css",
  ],
})
export class DriverInfoComponent implements OnInit {

  @ViewChild('removeDriverModal') removeDriverModal: ElementRef;
  /* @Input() eventsSubmit: Observable<void>;
  private eventsSubscription: Subscription; */
  @Output() scroll = new EventEmitter();

  driverInfoForm: FormGroup;

  hasChildren = false;
  stepNumber;
  stepsData;
  submitted: boolean;
  insuredWorkCityId: number;

  fromZeroToTen = DropDownValues.fromZeroToTen;
  isDrivingPercentageComplete: boolean;
  lang: string;

  months: { id: number; nameAr: string; name: string }[];
  years: number[] = [];

  previousQuote: QuoteRequest;
  previousQuoteDriver: Driver;
  previousQuoteDriverBirthDate: any;
  previousQuoteDriverBirthDateMonth: any;
  previousQuoteDriverBirthDateYear: any;
  termSearchFn = AppUtils.searchItemTerm;

  
  constructor(
    private insuranceSteps: InsuranceStepsService,
    private fb: FormBuilder,
    private scrollOnErrer: Scroll,
    private appService: AppService,
    private formvalidationService: FormsValidationService,
    private modal: NgbModal
  ) {}

  ngOnInit() {
    this.initForm();
    this.watchStepNum();
    this.watchDriversPercentage();
    // this.watchDrivingPercentage();
    this.appService.getAppLang().subscribe((language) => {
      this.lang = language;
    });
    this.watchIdentityNoToshowDate();
  }

  initForm() {
    this.driverInfoForm = this.fb.group({
      typeId: [1],
      genderId: ["M"],
      birthDate: [null],
      educationLevelId: [null, [Validators.required]],
      socialStatusId: [1],
      drivingPercentageId: [null, [Validators.required]],
      licenseTypeId: [null, [Validators.required]],
      workCityId: [null, [Validators.required]],
      childrenUnder16Years: [null],
      numberOfAccidentsLast5Years: [null],
      numberOfClaimsLast5Years: [null],
      month: [null, [Validators.required]],
      year: [null, [Validators.required]],
    });
    this.getStepData();
    this.getPreviousQuoteRequest();
    this.watchFormAfterSubmit();
  }

  get f() {
    return this.driverInfoForm.controls;
  }

  submitForm() {
    this.submitted = true;
    if (this.driverInfoForm.invalid) {
      const firstElementWithError = document.querySelector(
        ".ng-invalid[formControlName]"
      );
      this.scrollOnErrer.scrollTo(firstElementWithError);
      return;
    }

    const birthDate = {
      year: this.driverInfoForm.value.year,
      month: this.driverInfoForm.value.month,
    };

    const formattedDate = this.formatDate(birthDate);
    this.driverInfoForm.value.birthDate = formattedDate;
    const currentQuoteReq = this.insuranceSteps.getUserQuoteRequest().value;
    const mainDriver = this.driverInfoForm.value;
    mainDriver.identityNumber = currentQuoteReq.insuredIdentityNumber;
    // mainDriver.numberOfAccidents = currentQuoteReq.numberOfAccidents;
    mainDriver.homeCityId = null;

    const tempDrivers = [];

    tempDrivers.push(mainDriver);
    currentQuoteReq.insuredBirthDate = mainDriver.birthDate;
    currentQuoteReq.insuredGenderId = mainDriver.genderId;
    currentQuoteReq.insuredSocialStatusId = mainDriver.socialStatusId;
    currentQuoteReq.InsuredEducationLevelId = mainDriver.educationLevelId;
    currentQuoteReq.childrenUnder16Years = mainDriver.childrenUnder16Years;
    currentQuoteReq.insuredWorkCityId = mainDriver.workCityId;
    /* Check if previous quotation has drivers */
    if (currentQuoteReq.drivers) {
      /* loop throw driver  */

      currentQuoteReq.drivers.forEach((driver) => {
        if (driver.typeId == 1) {
          driver.genderId = mainDriver.genderId;
          driver.birthDate = mainDriver.birthDate;
          driver.educationLevelId = mainDriver.educationLevelId;
          driver.socialStatusId = mainDriver.socialStatusId;
          driver.drivingPercentageId = mainDriver.drivingPercentageId;
          driver.childrenUnder16Years = mainDriver.childrenUnder16Years;
          (driver.numberOfAccidentsLast5Years =
            mainDriver.numberOfAccidentsLast5Years),
            (driver.numberOfClaimsLast5Years =
              mainDriver.numberOfClaimsLast5Years);
        }
      });
    } else {
      mainDriver.medicalConditions = [];
      mainDriver.driverViolations = [];
      mainDriver.driverLicenses = new Array({
        licenseTypeId: this.driverInfoForm.value.licenseTypeId,
        countryId: 113,
        numberOfYears: 1,
        isSaudiLicense: true,
      });
      currentQuoteReq.drivers = tempDrivers;
    }

    // Remove the date from the form
    delete this.driverInfoForm.value.day;
    delete this.driverInfoForm.value.month;
    delete this.driverInfoForm.value.year;
    delete this.driverInfoForm.value.workCityId;
    this.insuranceSteps.overridenSetQuoteRequest(currentQuoteReq);
    if (this.stepNumber == 3) {
      this.insuranceSteps.setCurrentStepNum(4);
      this.formvalidationService.addFormsStatus({
        formName: FormsNames.driverInfoForm,
        isValid: true,
      });
    }

    // this.submitted = false;
  }

  formatDate(d): string {
    if (d === null) {
      return null;
    }

    return [d.month < 10 ? "0" + d.month : d.month, d.year].join("/");
  }

  watchDriversPercentage() {
    this.driverInfoForm
      .get("drivingPercentageId")
      .valueChanges.subscribe((val) => {
        const currentDriversPercentages: any[] = this.insuranceSteps.getUserQuoteRequest()
          .value.drivers;
        let mainDriverExists = false; // Indicates whether the main driver exists in the array
        let percentagesSum = 0; // Indicates the sum of all additional drivers percentages in the array

        currentDriversPercentages.forEach((driver) => {
          // If the main driver exists
          if (driver.typeId === 1) {
            mainDriverExists = true;
          }
          //  Calculate the sum of the additional drivers
          if (driver.typeId !== 1) {
            percentagesSum += driver.drivingPercentageId;
          }
        });
        // Add the value to be added to the percentages sum
        percentagesSum += val;
        if (mainDriverExists) {
          // Check if the selected percentage and previous percentages sum exceeds 100
          if (percentagesSum > 100) {
           
            // Show validation error
             this.f.drivingPercentageId.setErrors({
              drivingPercentageComplete: true,
            }); 
           
          } else {
          
             this.f.drivingPercentageId.setErrors(null); 

            // Update main driver with the new value
          }
        }
      });
  }

 
 
  getStepData() {
    this.insuranceSteps.getInsuranceStepsData().subscribe((stepsData) => {
      this.stepsData = stepsData;
    });
  }

  hasChild() {
    //    this.driverInfoForm.get("childrenUnder16Years").reset();
    this.hasChildren = !this.hasChildren;
    if (this.hasChildren) {
      this.driverInfoForm.controls["childrenUnder16Years"].setValidators([
        Validators.required,
      ]);
      this.driverInfoForm.controls[
        "childrenUnder16Years"
      ].updateValueAndValidity();
    } else {
      this.driverInfoForm.controls["childrenUnder16Years"].setValidators(null);
      this.driverInfoForm.controls[
        "childrenUnder16Years"
      ].updateValueAndValidity();
    }
  }

  watchStepNum() {
    this.insuranceSteps.getCurrentStepNum().subscribe((stepNum) => {
      this.stepNumber = stepNum;
    });
  }

  watchIdentityNoToshowDate() {
    var currentMeladiDate = new Date();
    var currentHijriYear: any = new Intl.DateTimeFormat("ar-TN-u-ca-islamic", {
      year: "numeric",
    }).format(Date.now());
    this.insuranceSteps.getUserQuoteRequest().subscribe((stepsData) => {
      // this.insuredId = stepsData.insuredIdentityNumber;
      if (stepsData.insuredIdentityNumber.toString().startsWith("2")) {
        this.months = DropDownValues.MeladiMonths;

        var currentYear = currentMeladiDate.getFullYear();
        var b = currentYear - 90;
        for (b; b <= currentYear - 17; b++) {
          this.years.push(b);
        }
      } else {
        this.months = DropDownValues.higriMonths;
        var b = parseInt(currentHijriYear.substring(0, 4)) - 90;
        for (b; b <= parseInt(currentHijriYear.substring(0, 4)) - 17; b++) {
          this.years.push(b);
        }
      }
    });
  }

  getPreviousQuoteRequest() {
    this.previousQuote = this.insuranceSteps.getUserQuoteRequest().value;
    this.f.workCityId.setValue(this.previousQuote.drivingCityId);

    if (this.previousQuote.drivers) {
      let driver = this.previousQuote.drivers.filter((drivers) => {
        return drivers.typeId == 1;
      })[0];

      this.previousQuoteDriver = driver;
      let year = driver.birthDate.substring(3, 7);
      let month = driver.birthDate.substring(0, 2);

      month.startsWith("0")
        ? this.f.month.setValue(parseInt(driver.birthDate.substring(1, 2)))
        : this.f.month.setValue(parseInt(month));
      this.f.year.setValue(parseInt(year));

      this.f.educationLevelId.setValue(driver.educationLevelId);
      this.f.drivingPercentageId.setValue(driver.drivingPercentageId);
      this.f.numberOfAccidentsLast5Years.setValue(
        driver.numberOfAccidentsLast5Years
      );
      this.f.numberOfClaimsLast5Years.setValue(driver.numberOfClaimsLast5Years);
      driver.driverLicenses.filter((license) => {
        if (license.isSaudiLicense) {
          this.f.licenseTypeId.setValue(license.licenseTypeId);
        }
      });
      this.previousQuoteDriver.socialStatusId == 2
        ? this.hasChild()
        : (this.hasChildren = false);
    }
  }

  watchFormAfterSubmit() {
    this.driverInfoForm.valueChanges.subscribe(() => {
      if (this.driverInfoForm.valid && this.stepNumber >= 4) {
        this.formvalidationService.addFormsStatus({
          formName: FormsNames.driverInfoForm,
          isValid: true,
        });
        this.submitForm();
      } else {
        this.formvalidationService.addFormsStatus({
          formName: FormsNames.driverInfoForm,
          isValid: false,
        });
      }
    });
  }
  /*  ngOnDestroy() {
    this.eventsSubscription.unsubscribe();
  } */
}
