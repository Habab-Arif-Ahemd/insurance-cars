import {
  Component,
  EventEmitter,
  OnInit,
  Output,
  OnDestroy,
  ChangeDetectionStrategy,
} from "@angular/core";
import { InsuranceStepsService } from "src/app/services/insurance-steps/insurance-steps.service";

@Component({
  changeDetection: ChangeDetectionStrategy.OnPush,
  selector: "app-driver-list",
  templateUrl: "./driver-list.component.html",
  styleUrls: [
    "./driver-list.component.css",
    "../../quotation-request-page/quotation-request-page.component.css",
  ],
})
export class DriverListComponent implements OnInit {
  @Output() addDriver = new EventEmitter();

  @Output() showAddDriver = new EventEmitter();

  currentDrivers;
  currentReq;
  inCompleteDrivingPercentage = false;
  percentageAlert = false;
  constructor(private insuranceService: InsuranceStepsService) {}

  ngOnInit() {
    this.getCurrentDriver();
  }

  getCurrentDriver() {
    this.insuranceService.getUserQuoteRequest().subscribe((reqData) => {
      reqData.drivers.sort((a, b) => (a.typeId > b. typeId)? 1 : -1);
      this.currentDrivers = reqData.drivers;
     /*  this.currentDrivers = this.currentDrivers.filter(
        (driver) => driver.typeId !== 1
      ); */
    });
  }

  removeDriver(driverToRemove) {
    const currentReq = this.insuranceService.userQuoteRequest.value;
    let currentDrivers = currentReq.drivers;
    currentDrivers = currentDrivers.filter(
      (driver) => driver !== driverToRemove
    );
    currentReq.drivers = currentDrivers.concat();
    this.insuranceService.overridenSetQuoteRequest(currentReq);
  }

  submitAddDriver() {
    let percentagesSum = 0; // Indicates the sum of all additional drivers percentages in the array
    this.insuranceService
      .getUserQuoteRequest()
      .value.drivers.forEach((driver) => {
        //  Calculate the sum of the  drivers
        percentagesSum += driver.drivingPercentageId;
      });

    if (percentagesSum == 100) {
      this.inCompleteDrivingPercentage = true;
      this.percentageAlert = false;

    } else {
      this.inCompleteDrivingPercentage = false;

      // show add new driver form
      this.showAddDriver.emit();

      // send event to open the form
      this.addDriver.emit();
    }
  }
  
 
}
