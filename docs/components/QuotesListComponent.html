import { QuoteService } from 'src/app/services/quote/quote.service';
import { Component, OnInit } from '@angular/core';
import { Quote } from '../../../models/quote/Quote';
import { Product } from 'src/app/models/quote/Product';
import { LocallyStoredItemsKeys } from 'src/app/models/app/LocallyStoredItemsKeys';


@Component({
   selector: 'app-quotes-list',
   templateUrl: './quotes-list.component.html',
   styleUrls: ['./quotes-list.component.css']
})


export class QuotesListComponent implements OnInit {

   quotes: Quote[] = [];

   constructor(private quoteService: QuoteService) { }

   ngOnInit() {
      // Get quotes list from local storage
      this.quoteService.getQuotes().subscribe(quotes => {
         this.quotes = quotes;
      });
   }

}

<div *ngIf="quotes.length === 0" class="flex-container">

</div>

</div>
