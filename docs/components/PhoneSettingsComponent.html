import { Component, OnInit } from '@angular/core';
import { AuthService } from 'src/app/services/auth/auth.service';
import { FormGroup, Validators, FormControl } from '@angular/forms';
import { UserService } from 'src/app/services/user/user.service';
import { slideIn } from 'src/app/helpers/utils/animations';
import { Router, ActivatedRoute } from '@angular/router';
import { Location } from '@angular/common';
import { AppRoutes } from 'src/app/models/app/AppRoutes';
import { ApiResponse } from 'src/app/payload/responses/ApiResponse';
import { PhoneChangeRequest } from 'src/app/payload/requests/user/PhoneChangeReq';
import { PhoneChangeVerificationReq } from 'src/app/payload/requests/user/PhoneChangeVerificationReq';


@Component({
   selector: 'app-phone-settings',
   templateUrl: './phone-settings.component.html',
   styleUrls: ['./phone-settings.component.css'],
   animations: [ slideIn ]
})


export class PhoneSettingsComponent implements OnInit {

   /* Forms */
   phoneSettingsForm: FormGroup;

   /* Data */
   userPhoneNumber: string;

   /* UI */
   isLoading: boolean;
   isActive: boolean;
   isVerifying: boolean;
   isSubmitting: boolean;

   /* Error */
   isErrorAlertVisible: boolean;
   errorMessage: string;


   constructor(
      private authService: AuthService, private userService: UserService,
      private activeRoute: ActivatedRoute, private router: Router,
      private location: Location
   ) { }


   ngOnInit() {
      this.initForm();
      this.getUserData();
      this.watchActiveSettingsPanel();
      this.watchSettingsRoute();
   }


   /* INITIALIZATION */

   private initForm(): void {
      this.phoneSettingsForm = new FormGroup({
         phoneNumber: new FormControl(null, [Validators.required, Validators.minLength(9)]),
         password: new FormControl(null, [Validators.required, Validators.minLength(9)]),
         verificationCode: new FormControl()
      });
   }

   get phoneForm() {
      return this.phoneSettingsForm.controls;
   }

   private getUserData(): void {
      this.userPhoneNumber = this.userService.getUserProfileData().phone;
   }

   private watchActiveSettingsPanel(): void {
      this.userService.getActiveSettingsPanel().subscribe(panels => {
         if (panels.phone) {
            this.isActive = true;
         } else {
            this.isActive = false;
         }
      });
   }

   private watchSettingsRoute(): void {
      this.activeRoute.params.subscribe(param => {
         if (param.activePanel === 'phone') {
            this.isActive = true;
            this.isVerifying = false;
         }
         if (param.state === 'verify') {
            this.isVerifying = true;
         }
      });
   }



   /* UI & ROUTING */

   cancelEdit(): void {
      this.userService.resetActiveSettingsPanel();
      this.router.navigate([AppRoutes.controlPanel.accountSettings.uri]);
   }

   setActive(): void {
      // Set current panel as active
      this.userService.setActiveSettingsPanel({ password: false, email: false, phone: true, misc: false });
      this.router.navigate([AppRoutes.controlPanel.accountSettings.phone.main]);
   }



   /* FORMS SUBMIT */

   submitPhoneChange(): void {

      // Display loader
      this.isLoading = true;
      this.isSubmitting = true;

      // Validate inputs
      if (this.phoneSettingsForm.invalid) {
         this.isLoading = false;
         return
      }

      // Construct the phone change request
      const phoneChangeReq: PhoneChangeRequest = {
         userId:this.authService.getTokenUserId(),
         newPhoneNumber: this.phoneForm.phoneNumber.value,
         password: this.phoneForm.password.value,
      };

      // Send the request
      this.userService.requestPhoneNumberChange(phoneChangeReq).subscribe(
         (res: ApiResponse) => {
            if (res.isSuccess) {

               // Stop loader/submitting
               this.isLoading = false;
               this.isSubmitting = false;

               // Change URL to '/verify' without navigating to keep UI state on refresh
               this.location.go(AppRoutes.controlPanel.accountSettings.phone.verify);

               // Display the verification code input
               this.isVerifying = true;

            }
         },
         error => this.showErrorAlert(error)
      );

   }

   submitPhoneChangeVerification(): void {

      const newPhoneNumber = this.phoneForm.phoneNumber.value;
      const verificationCode = this.phoneForm.verificationCode.value;

      // Display loader
      this.isLoading = true;
      this.isSubmitting = true;

      // Set the password control value so it validates (since it could be reset on refresh. It will be hidden, and its value will not be used)
      this.phoneForm.password.setValue('0');

      // Validate inputs and verification code
      if (!newPhoneNumber || !verificationCode) {
         this.isLoading = false;
         return
      }

      // Construct the change verification request
      const phoneChangeVerificationReq: PhoneChangeVerificationReq = {
         userId: this.authService.getTokenUserId(),
         newPhoneNumber: newPhoneNumber,
         verificationCode: verificationCode
      };

      // Send the request
      this.userService.changePhoneNumber(phoneChangeVerificationReq).subscribe(
         (res: ApiResponse) => {
            if (res.isSuccess) {

               // Stop loader/submitting
               this.isLoading = false;
               this.isSubmitting = false;

               // Hide the verification code input
               this.isVerifying = false;

               // Update the user's phone on the stored user profile data
               const profileData = this.userService.getUserProfileData();
               profileData.email = newPhoneNumber;
               this.userService.setUserProfileData(profileData);

               // TODO: Show success message/art
               
            }
         },
         error => this.showErrorAlert(error)
      );

   }



   /* MISC */

   private showErrorAlert(errorResponse: any): void {
      // Get error data
      const error: ApiResponse = errorResponse.error;

      // Stop the loader
      this.isLoading = false;
      this.isSubmitting = false;

      // Get error message
      if (error.validationErrors && error.validationErrors.length > 0) {
         this.errorMessage = error.validationErrors[0].description;
      } else if (!error.validationErrors && error.responseMessage) {
         this.errorMessage = error.responseMessage;
      } else {
         this.errorMessage = 'An error has occurred';
      }

      // Show error message
      if (this.errorMessage.length > 0) {
         this.isErrorAlertVisible = true;
      }
   }

}

            <span>{{ isVerifying ? ('Misc.SendVerificationCodeText' | translate) : ('App.SaveChangesText' | translate) }}</span>
