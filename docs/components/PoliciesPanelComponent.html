import { Component, OnInit } from '@angular/core';
import { ProfileService } from 'src/app/services/profile/profile.service';
import { DashboardResponse } from 'src/app/models/order-preview/DashboardResponse';


@Component({
   selector: 'app-policies-panel',
   templateUrl: './policies-panel.component.html',
   styleUrls: ['./policies-panel.component.css']
})


export class PoliciesPanelComponent implements OnInit {

   polciesColors = {
      bought: '#00a2ae',
      expiring: '#e3b87b',
      expired: '#f75959',
      uncompleted: '#7eaced'
   };

   userPolicies: any[] = [
      /*
      { insuranceCompany: 'شركة 01', id: 653298, status: 'سارية', initiationDate: '10/01/2020', startDate: '09/02/2020', endDate: '12/05/2021'},
      { insuranceCompany: 'شركة 02', id: 357951, status: 'سارية', initiationDate: '10/01/2020', startDate: '09/02/2020', endDate: '12/05/2021'},
      */
   ];

   isLoadingPoliciesAnalytics = false;
   isLoadingUserPolicies = false;
   policiesAnalytics: DashboardResponse;


   constructor(private profileService: ProfileService) { }


   ngOnInit() {
      this.getDashboardData();
    }

   getDashboardData() {
      this.isLoadingPoliciesAnalytics = true;
      this.profileService.getDashboardData().subscribe(policiesAnalytics => {
         this.policiesAnalytics = policiesAnalytics
         this.isLoadingPoliciesAnalytics = false;
         console.log(this.policiesAnalytics)
      })
   }

}
