   /**
    * Retrieves the preview quote / checkout data
    */
   private getPreviewQuote(): void {
      this.orderedQuote = this.quoteService.getPreviewQuote();
      // Sort driver array to start with main driver
      this.orderedQuote.vehicleDrivers.sort((a, b) => (a.typeId > b. typeId)? 1 : -1);
      this.orderedQuote.quoteInfo.isSelectedForOrdering = true;
      this.orderedQuote.quoteInfo.product.isSelected = true;
      this.orderedQuote.quoteInfo.product.hasExtraBenefits = true;
      this.orderedQuote.quoteInfo.product.benefits.forEach(benefit => {
         if (benefit.benefitTypeId === 3) {
            benefit.isSelected = true;
            benefit.isChecked = true;
         }
      });
   }


   /**
    * Initializes the vehicle images uploader and handles the uploads
    */
   private initUploader(): void {

      /* Uploader options */
      this.uploader = new FileUploader({

         /* Request Opts */
         url: environment.apiUrl + ApiRoutes.Quotation.PreviewAttachment,
         headers: [{ name: 'Accept', value: 'application/json' }],
         autoUpload: true,

         /* File Opts */
         allowedMimeType: ['image/png', 'image/jpeg', 'image/gif', 'image/tiff', 'image/bmp'],
         allowedFileType: ['image'],
         maxFileSize: 30000000 // 30MB

      });

      this.uploader.onWhenAddingFileFailed = () => {
         this.activeAttachmentUpload.isInvalidFile = true;
         this.checkInvalidFileUpload();
      };
      
      this.uploader.onAfterAddingFile = () => {
         this.activeAttachmentUpload.isInvalidFile = false;
         this.checkInvalidFileUpload();
      };

      /* Form data */
      this.uploader.onBuildItemForm = (item, form) => {
         form.append('UserId', this.authService.getTokenUserId());
         form.append('ReferenceId', this.orderedQuote.referenceId);
      };

      /* Before upload options */
      this.uploader.onBeforeUploadItem = item => {
         item.withCredentials = false;
         item.alias = 'Attachment';
         item.file.name = this.activeAttachmentUpload.title + "." + item._file.name.split('.').pop();
      };

      /* Upload progress */
      this.uploader.onProgressItem = (item, progress) => {
         if (progress < 100) {
            this.activeAttachmentUpload.isUploading = true;
            this.activeAttachmentUpload.uploadProgress = progress;
         }
         item.onComplete = (res) => {
            this.activeAttachmentUpload.isUploading = false;
         }
      };

      /* Success callback */
      this.uploader.onSuccessItem = (item, response, status) => {

         // Get response object
         let uploadRes: OrderAttachmentUploadResponse = JSON.parse(response);

         // Construct image source url to view the image
         this.activeAttachmentUpload.imageUrl = 
         environment.apiUrl + 
         ApiRoutes.Quotation.PreviewAttachment + 
         '?userId=' + uploadRes.userId +
         '&referenceId=' + uploadRes.referenceId +
         '&attachmentFileName='+ uploadRes.attachmentFileName;

         // Update uploaded status and display the image
         this.activeAttachmentUpload.isUploaded = true;

      };

      /* Error callback */
      this.uploader.onErrorItem = (item, response, status) => {};

   }

   private renderPaymentform() {

      // Watch payment form inputs change when form is loaded
      /* hyperpayModule.paymentFormReady().then(() => {

         console.log('Getting card holder input');

         let cardNumIpt = document.getElementsByName("card.holder")[0];
         cardNumIpt.addEventListener('change', (event) => {

            console.log(event);
         });
         cardNumIpt.oninput = (val) => {
            console.log('On Ipt');
            console.log(val);
         };

      }); */

      this.appService.getAppLang().subscribe(async (lang) => {

         let hyperpayOptsScriptId = 'payment-form-options';
         let hyperpayScriptId = 'hyperpay-script';
         this.checkoutId = this.checkoutId;

         /* ------------------------- Create HyperPay Scripts ------------------------ */

         // Create new HyperPay payment script
         const newHyperpayScript = document.createElement("script");
         newHyperpayScript.id = hyperpayScriptId;
         newHyperpayScript.src = `https://test.oppwa.com/v1/paymentWidgets.js?checkoutId=${this.checkoutId}`;
         newHyperpayScript.async = true;

         // Create new HyperPay options script with options
         const optionsScript = document.createElement("script");
         optionsScript.id = hyperpayOptsScriptId;
         optionsScript.type = 'text/javascript';
         optionsScript.innerHTML = `
            var wpwlOptions = {
               style: "card",
               locale: "${lang}",
               brandDetection: true,
               brandDetectionType: "binlist",
               brandDetectionPriority: ["MADA","VISA","MASTER"],
               createRegistration: false,
               onReady: () => hyperpayModule.paymentFormReady().resolve()
            };
         `;
      
         /* ----------------- Replace old options script if it exists ---------------- */
         
         // Get old options script
         const oldOptsScript = document.getElementById(hyperpayOptsScriptId);
         
         // If it exists
         if (oldOptsScript) {
      
            // Remove the hyperpay payment form from DOM 
            this.isCardPaymentFormVisible = false;
      
            await AppUtils.delay(100);
      
            // Display and add the hyperpay payment form to DOM again
            this.isCardPaymentFormVisible = true;
      
            // Replace options script with new language
            document.head.replaceChild(optionsScript, oldOptsScript);
            
         } else {
            // Append it
            document.head.appendChild(optionsScript);
         }
      
      
         /* ----------------- Reload old hyperpay script if it exists ---------------- */
      
         // Get the old hyperpay script
         const oldHyperpayScript = document.getElementById(hyperpayScriptId);
         
         // If it exists
         if (oldHyperpayScript) {
            // Reload it
            document.head.removeChild(oldHyperpayScript);
            document.head.appendChild(newHyperpayScript);
         } else {
            // Append it
            document.head.appendChild(newHyperpayScript);
         }
            
      });

   }

   openContinuationModal(modal: any) {
      // If closed
      if (!this.isContinuationModalDisplayed) {
         // Open the modal
         this.modalService.open(modal, {
            centered: true, beforeDismiss: () => {
               // Switch flag when dissmissed to uncheck the checkbox
               this.isContinuationModalDisplayed = false;
               // Close the modal
               return true;
            },
            size: 'lg'
         });
         // Set flag to check the checkbox
         this.isContinuationModalDisplayed = true;
      }
   }


   getAppLang() {
      this.appService.getAppLang().subscribe(lang => (this.lang = lang));
   }


   toggleAccordion(accordion: 'Drivers' | 'ImageUpload' | 'Payment' | 'Card' | 'Sadad') {
      switch (accordion) {
         case 'Drivers':
            this.accordionsState.isDriversDisplayed = !this.accordionsState.isDriversDisplayed;
            break;
         case 'ImageUpload':
            this.accordionsState.isImageUploadDisplayed = !this.accordionsState.isImageUploadDisplayed;
            break;
         case 'Payment':
            this.accordionsState.isPaymentDisplayed = !this.accordionsState.isPaymentDisplayed;
            break;
         case 'Card':
            this.accordionsState.isCardPaymentDisplayed = !this.accordionsState.isCardPaymentDisplayed;
            break;
         case 'Sadad':
            this.accordionsState.isSadadPaymentDisplayed = !this.accordionsState.isSadadPaymentDisplayed;
            break;
         default: break;
      }
   }


   setActiveAttachmentUpload(attachment: PolicyOrderAttachment) {

   checkInvalidFileUpload() {
      let invalidFiles = this.orderFiles.filter(file => file.isInvalidFile);
      if (invalidFiles.length > 0) this.isInvalidFileAlertDisplayed = true;
      else this.isInvalidFileAlertDisplayed = false;
   }

   saveOrderProgress() {

      // Display loader

      let reqBody:  SavedQuoteRequest = {quoteId: this.orderedQuote.referenceId}; 
     // reqBody.quoteId = this.orderedQuote.referenceId;

         (res) => {
            console.log(res);
            this.modalService.dismissAll();
            this.router.navigate(['/user/previous-quotes']);

         },
         (err) => {
            console.log(err);
            // Stop loader
            this.isSavingOrder = false;
            // TODO: Handle error
         }
      );

   }
