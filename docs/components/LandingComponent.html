import { Component, OnInit, ViewChild, ElementRef } from '@angular/core';
import { Router } from '@angular/router';
import { InsuranceStepsService } from 'src/app/services/insurance-steps/insurance-steps.service';
import { FormGroup, FormBuilder, Validators } from '@angular/forms';
import { PreliminaryDataRequest } from 'src/app/payload/requests/quotation/PreliminaryDataRequest';
import { InsuranceStepsApiService } from 'src/app/services/insurance-steps/insurance-steps-api.service';
import AppUtils from 'src/app/helpers/utils/AppUtils';
import { AppRoutes } from 'src/app/models/app/AppRoutes';
import { AuthService } from 'src/app/services/auth/auth.service';
import { NationalIdValidator } from 'src/app/helpers/validators/NationalIdValidator';
import { LocallyStoredItemsKeys } from 'src/app/models/app/LocallyStoredItemsKeys';


@Component({
   selector: 'app-landing',
   templateUrl: './landing.component.html',
   styleUrls: ['./landing.component.css']
})


export class LandingComponent implements OnInit {

   /* Captcha */
   @ViewChild('captchacanvas', { static: true }) captchaCanvas: ElementRef;
   private canvasContext: CanvasRenderingContext2D;
   captchaCode;

   /* Form */
   landingFormGroup: FormGroup;
   isLoading = false;
   isFormValid = false;
   errorMessage: string;
   isAgree= false;

   /* Carousel */
   @ViewChild('ngbCarousel') landingCarousel: ElementRef;
   images = [
      'https://images.unsplash.com/photo-1498887960847-2a5e46312788?ixlib=rb-1.2.1&ixid=eyJhcHBfaWQiOjEyMDd9&auto=format&fit=crop&w=1000&h=1000&q=80',
      'https://images.unsplash.com/photo-1470162015499-2b9772941cde?ixlib=rb-1.2.1&auto=format&fit=crop&w=1000&h=1000&q=80',
      'https://images.unsplash.com/photo-1471174617910-3e9c04f58ff5?ixlib=rb-1.2.1&ixid=eyJhcHBfaWQiOjEyMDd9&auto=format&fit=crop&w=1000&h=1000&q=80'
   ];
   showSequenceNo= true;
   showCardNo: boolean;


   constructor(
      private router: Router, private fb: FormBuilder, private authService: AuthService,
      private insuranceStepsService: InsuranceStepsService, private insuranceStepsApi: InsuranceStepsApiService
   ) { }


   ngOnInit() {
      this.initForm();
      this.onChangeRegistration();
      this.initCaptchaCanvas();
      this.watchFormChanges();
      this.initInsuranceSteps();
      this.removePreviousQuotaionData();
   }

   removePreviousQuotaionData() {
     let previousQuote = localStorage.getItem(LocallyStoredItemsKeys.UserQuoteRequest);
   
      if (previousQuote) {
         localStorage.removeItem(LocallyStoredItemsKeys.UserQuoteRequest);
      } 

   }
   initForm(): void {
      this.landingFormGroup = this.fb.group({
         insuredIdentityNumber: ['', Validators.compose([
            Validators.required,
            NationalIdValidator,
            Validators.minLength(10)
         ])],
         vehicleId: ['', Validators.compose([
            Validators.required,
            Validators.minLength(10)
         ])],
         vehicleIdTypeId: [1 , [Validators.required]],
         captcha: ['' , [Validators.required]],
      });

      /* Submit on enter press */
      document.getElementById('landing-form-box').addEventListener('keyup', (e) => {
         if (e.code === 'NumpadEnter' || e.code === 'Enter') {
            // Trigger the button element with a click
            this.onSubmit();
         }
      });
   }


   get landingForm() { return this.landingFormGroup.controls; }


   initInsuranceSteps() {
      // Reset insurance steps counter
      this.insuranceStepsService.setCurrentStepNum(0);
   }


   async initCaptchaCanvas() {
      this.initCaptchaContent();
      await AppUtils.delay(500);
      this.initCaptchaContent();
   }


   initCaptchaContent() {
      const charsArray = '0123456789';
      const lengthOtp = 6;
      const captcha = [];
      const canv = this.captchaCanvas.nativeElement;
      this.canvasContext = (canv as HTMLCanvasElement).getContext('2d');

      // Clear canvas
      this.canvasContext.clearRect(0, 0, this.canvasContext.canvas.width, this.canvasContext.canvas.height);

      for (let i = 0; i < lengthOtp; i++) {
         const index = Math.floor(Math.random() * charsArray.length + 1);
         if (captcha.indexOf(charsArray[index]) === -1) {
            captcha.push(charsArray[index]);
         } else { i--; }
      }

      this.captchaCode = captcha.join('');

      this.canvasContext.font = '16px Comfortaa';
      this.canvasContext.textBaseline = 'middle';
      this.canvasContext.textAlign = 'center';

      const x = (canv as HTMLCanvasElement).width / 2;
      const y = (canv as HTMLCanvasElement).height / 2;
      this.canvasContext.strokeText(this.captchaCode, x, y);

      // Revalidate form with new captcha
      this.validateOrderForm();
   }


   isCaptchaValid(): boolean {
      return (this.landingFormGroup.get('captcha').value === this.captchaCode);
   }


   watchFormChanges() {
      this.landingFormGroup.valueChanges.subscribe(() => {
         this.validateOrderForm();
      });
   }


   validateOrderForm() {
      if (this.isCaptchaValid() && this.landingForm.insuredIdentityNumber.valid && this.landingForm.vehicleId.valid) {
         this.isFormValid = true;
      } else {
         this.isFormValid = false;
      }
   }


   onSubmit() {

      // Show loading spinner
      this.isLoading = true;

      // Validate form
      if (!this.isFormValid) {
         this.isLoading = false;
         return;
      }

      // Remove the captcha from the form
      delete this.landingFormGroup.value.captcha;

      // Construct request data which is used to retreive the preliminary quote order data
      const preliminaryDataRequest: PreliminaryDataRequest = {
         insuredIdentityTypeId: parseInt(this.landingFormGroup.value.insuredIdentityNumber.slice(0, 1), 10),
         insuredIdentityNumber: parseInt(this.landingFormGroup.value.insuredIdentityNumber, 10),
         vehicleIdTypeId: parseInt(this.landingFormGroup.value.vehicleIdTypeId, 10),
         vehicleId: parseInt(this.landingFormGroup.value.vehicleId, 10)
      };

      // Add the data to the quotation request construction object
      this.insuranceStepsService.setUserQuoteRequest(preliminaryDataRequest);

      // Attempt to get the user's previous quotation request preliminary data
      this.getUserPreliminaryData(preliminaryDataRequest);

   }


   private getUserPreliminaryData(preliminaryDataRequest: PreliminaryDataRequest): void {
      // Send the request
      this.insuranceStepsApi.getUserQuotePreliminaryData(preliminaryDataRequest).subscribe(
         preliminaryData => {

            // Add the prelimary data to the quotation request construction object
            this.insuranceStepsService.setUserQuoteRequest(preliminaryData);

            // Navigate the user
            this.navigateUser();

            // Hide spinner
            this.isLoading = false;

         },
         err => {
            // If preliminary data was not found
            if (err.error.statusCode === 404) {
               // Navigate the user anyway
               this.navigateUser();
            } else {
               // Show error message
               this.errorMessage = err.error.responseMessage;
            }

            // Hide spinner
            this.isLoading = false;

         }
      );
   }
   onChangeRegistration() {
      this.landingForm.vehicleIdTypeId.valueChanges.subscribe(vehicleIdTypeId => {
         if (vehicleIdTypeId == 1) {
            this.showSequenceNo = true;
            this.showCardNo = false;
         } else {
            this.showCardNo = true
            this.showSequenceNo = false
         }
      })
   }

   private navigateUser() {
      // Route to the insurance order forms page
      this.router.navigate([AppRoutes.insuranceSteps]);
      // Increase insurance steps counter
      this.insuranceStepsService.setCurrentStepNum(1);
   }

}
