import { Component, EventEmitter, HostListener, Input, OnInit, Output, ChangeDetectionStrategy } from "@angular/core";
import { FormBuilder, FormGroup, Validators } from "@angular/forms";
import DropDownValues from "src/app/helpers/utils/DropDownValues";
import { Scroll } from "src/app/helpers/utils/scroll";
import { NationalIdValidator } from "src/app/helpers/validators/NationalIdValidator";
import { DriverLicense } from "src/app/models/insurance-steps/DriverLicense";
import { AppService } from "src/app/services/app.service";
import { InsuranceStepsService } from "src/app/services/insurance-steps/insurance-steps.service";
import { debounceTime, distinctUntilChanged } from 'rxjs/operators';
import { Driver } from 'src/app/models/insurance-steps/Driver';
import AppUtils from 'src/app/helpers/utils/AppUtils';

@Component({
  changeDetection: ChangeDetectionStrategy.Default,
  selector: "app-additional-driver",
  templateUrl: "./additional-driver.component.html",
  styleUrls: [
    "../../quotation-request-page/quotation-request-page.component.css",
    "./additional-driver.component.css",
  ],
})
export class AdditionalDriverComponent implements OnInit {
  @Output() showList = new EventEmitter();

  // event comming from diver component
  @Input() openAddingDriverForm: boolean;

  additionalDriverInfoForm: FormGroup;
  driverLicenseForm: FormGroup;

  isDrivingPercentageComplete: boolean;

  driverLicense = false;
  driverViolation = false;
  driverHealth = false;
  addedlicense = false;
  addingDriver = false;

  hasChildren = false;
  stepNumber: any;
  stepsData;

  fromZeroToTen = DropDownValues.fromZeroToTen;
  showDate = false;
  months: { id: number; nameAr: string; name: string }[];
  years: number[] = [];

  licenseslist = [];
  foreignlicenses = [];
  medicalConditions: any = [];
  violations = [];

  additionalDriverInfoFormSubmitted: boolean;
  driverLicenseFormSubmitted: boolean;
  isLoading: boolean;
  isMedicalCheckboxEmpty: boolean;
  isViolationCheckboxEmpty: boolean;
  lang: string;
  isOnMidScreen: boolean;

  day: number[] = [];

  hijriYears: number[] = [];

  showGregorianDate = false;
  showHijriDate = false;
  monthvalue: number;
  previousDrivers: Driver[]= [];
  termSearchFn = AppUtils.searchItemTerm;

  constructor(
    private fb: FormBuilder,
    private insuranceSteps: InsuranceStepsService,
    private scrollOnErrer: Scroll,
    private appService: AppService
  ) {}

  ngOnInit() {
    this.getStepData();
    if (this.openAddingDriverForm) {
      this.showAditionalDriver();
    }
    this.appService.getAppLang().subscribe((language) => {
      this.lang = language;
    });
    

  }

  initForm() {
    this.additionalDriverInfoForm = this.fb.group({
      typeId: [2],
      genderId: ["M"],
      identityNumber: [
        "",
        Validators.compose([
          Validators.required,
          NationalIdValidator,
          Validators.minLength(10),
        ]),
      ],
      birthDate: [null],
      educationLevelId: [null, [Validators.required]],
      socialStatusId: [1],
      drivingPercentageId: [null, [Validators.required]],
      homeCityId: [null],
      relationId: [null, [Validators.required]],
      licenseTypeId: [null, [Validators.required]],
      numberOfAccidentsLast5Years: [null],
      numberOfClaimsLast5Years: [null],
      childrenUnder16Years: [null],
      medicalConditions: [this.medicalConditions],
      driverViolations: [this.violations],
      driverLicenses: [this.foreignlicenses],
      month: [null, [Validators.required]],
      year: [null, [Validators.required]],
    });
    this.whatchDriversPercentage();
    this.initdriverLicensesForm();
    this.validateIdentityNo();
    this.watchIdentityNoToShowDate();
  }

  showAditionalDriver() {
    this.addingDriver = !this.addingDriver;
    this.additionalDriverInfoFormSubmitted = false;
    this.initForm();
    
  }
  cancelAddingDriver() {
    this.addingDriver = !this.addingDriver;
    this.foreignlicenses.length = 0;

    const firstElementWithError = document.querySelector("#add-driver");
    this.scrollOnErrer.scrollTo(firstElementWithError);
    return;
  }

  whatchDriversPercentage() {
    this.additionalDriverInfoForm
      .get("drivingPercentageId")
      .valueChanges.subscribe((percentage) => {
        let previousDrivers = this.insuranceSteps.userQuoteRequest.value.drivers;
        let percentagesSum = 0; // Indicates the sum of all additional drivers percentages in the array

        previousDrivers.forEach((driver) => {
          // If the main driver exist
            percentagesSum += driver.drivingPercentageId;
        });
        // Add the value to be added to the percentages sum
        percentagesSum += percentage;

          // Check if the selected percentage and previous percentages sum exceeds 100
          if (percentagesSum > 100) {
  
            // Show validation error
             this.f.drivingPercentageId.setErrors({
              drivingPercentageComplete: true,
            }); 
          } else {
             this.f.drivingPercentageId.setErrors(null); 

            // Update main driver with the new value
          }
        
      });
 
  }

  // convenience getter for easy access to form fields f stand for additional driver form and df for drive license form
  get f() {
    return this.additionalDriverInfoForm.controls;
  }
  get df() {
    return this.driverLicenseForm.controls;
  }

  formatDate(d): string {
    if (d === null) {
      return null;
    }

    return [d.month < 10 ? "0" + d.month : d.month, d.year].join("/");
  }

  submitForm() {
    this.additionalDriverInfoFormSubmitted = true;
    this.validateCheckBoxes();
    if (this.additionalDriverInfoForm.invalid) {
      const firstElementWithError = document.querySelector(
        ".ng-invalid[formControlName]"
      );
      this.scrollOnErrer.scrollTo(firstElementWithError);
      return;
    }

    const birthDate = {
      year: this.additionalDriverInfoForm.value.year,
      month: this.additionalDriverInfoForm.value.month,
    };

    const formattedDate = this.formatDate(birthDate);
    this.additionalDriverInfoForm.value.birthDate = formattedDate;

    this.additionalDriverInfoForm.value.childrenUnder16Years =
      this.additionalDriverInfoForm.value.insuredSocialStatusId === 2
        ? this.additionalDriverInfoForm.value.childrenUnder16Years
        : null;
    const driverInfo = this.additionalDriverInfoForm.value;
    driverInfo.identityNumber = parseInt(
      this.additionalDriverInfoForm.value.identityNumber
    );
    const tempDrivers = [];
    tempDrivers.push(driverInfo);
    // Remove the date from the form
    delete this.additionalDriverInfoForm.value.day;
    delete this.additionalDriverInfoForm.value.month;
    delete this.additionalDriverInfoForm.value.year;
    this.insuranceSteps.setUserQuoteReqDrivers(tempDrivers);
    this.additionalDriverInfoFormSubmitted = false;
    
    this.showList.emit();
  }

  initdriverLicensesForm() {
    this.driverLicenseForm = this.fb.group({
      licenseTypeId: [null],
      countryId: [null],
      numberOfYears: [null],
    });
  }

  getStepData() {
    this.insuranceSteps.getInsuranceStepsData().subscribe((stepsData) => {
      this.stepsData = stepsData;
    });
  }

  addLicense(id: DriverLicense) {
    this.driverLicenseFormSubmitted = true;
    if (this.driverLicenseForm.invalid) {
      return;
    }
    const country = this.stepsData.Countries.filter(
      (country) => country.id === id.countryId
    )[0];
    const licenseType = this.stepsData.LicenseTypes.filter(
      (type) => type.id === id.licenseTypeId
    )[0];
    const noOfYear = this.fromZeroToTen.filter(
      (no) => no.id === id.numberOfYears
    )[0];
    const driverLicense = {
      country: country,
      licenseType: licenseType,
      noOfYear: noOfYear,
    };
    this.licenseslist.push(driverLicense);
    this.foreignlicenses.push(id);
    this.driverLicenseForm.reset();
    this.driverLicenseFormSubmitted = false;
  }

  removeFromLicenseList(licenseData: any): void {
    // Remove from licenses list
    const index: number = this.licenseslist.indexOf(licenseData);
    if (index !== -1) {
      this.licenseslist.splice(index, 1);
    }

    // Construct foreign license object to be removed from the array
    const licenseToRemove = {
      driverLicenses: {
        licenseTypeId: licenseData.licenseType.id,
        countryId: licenseData.country.id,
        numberOfYears: licenseData.noOfYear.id,
      },
    };

    // Find the index of the foreign license to be removed in the foreign licenses array
    let foreignLicenseIdx = -1;
    for (let i = 0; i < this.foreignlicenses.length; i++) {
      if (
        JSON.stringify(this.foreignlicenses[i]) ===
        JSON.stringify(licenseToRemove)
      ) {
        foreignLicenseIdx = i;
      }
    }

    // Remove the license from the foreign licenses array
    if (foreignLicenseIdx !== -1) {
      this.foreignlicenses.splice(foreignLicenseIdx, 1);
    }
  }

  addViolation(id: number) {
    this.isViolationCheckboxEmpty = false;
    var index = this.violations
      .map((e) => {
        return e.violationId;
      })
      .indexOf(id);
    if (index === -1) {
      this.violations.push({ violationId: id });
    } else {
      this.violations.splice(index, 1);
    }
  }

  addMedicalCondition(id: number) {
    this.isMedicalCheckboxEmpty = false;
    var index = this.medicalConditions
      .map((e) => {
        return e.medicalConditionId;
      })
      .indexOf(id);
    if (index === -1) {
      this.medicalConditions.push({ medicalConditionId: id });
    } else {
      this.medicalConditions.splice(index, 1);
    }
  }

  hasChild() {
    this.additionalDriverInfoForm.get("childrenUnder16Years").reset();
    this.hasChildren = !this.hasChildren;

    if (this.hasChildren) {
      this.additionalDriverInfoForm.controls[
        "childrenUnder16Years"
      ].setValidators([Validators.required]);
      this.additionalDriverInfoForm.controls[
        "childrenUnder16Years"
      ].updateValueAndValidity();
    } else {
      this.additionalDriverInfoForm.controls[
        "childrenUnder16Years"
      ].setValidators(null);
      this.additionalDriverInfoForm.controls[
        "childrenUnder16Years"
      ].updateValueAndValidity();
    }
  }
  showLicenseForm() {
    this.driverLicense = !this.driverLicense;
    this.driverLicenseFormSubmitted = false;
    this.driverLicenseForm.reset();
    this.foreignlicenses.length = 0;
    this.licenseslist.length = 0;

    if (this.driverLicense) {
      this.driverLicenseForm.controls["licenseTypeId"].setValidators([
        Validators.required,
      ]);

      this.driverLicenseForm.controls["countryId"].setValidators([
        Validators.required,
      ]);
      this.driverLicenseForm.controls["numberOfYears"].setValidators([
        Validators.required,
      ]);
      this.driverLicenseForm.updateValueAndValidity();
    } else {
      this.driverLicenseForm.controls["licenseTypeId"].clearValidators();
      this.driverLicenseForm.controls["countryId"].clearValidators();
      this.driverLicenseForm.controls["numberOfYears"].clearValidators();
      this.driverLicenseForm.updateValueAndValidity();
    }
  }

  showMedicalCondition() {
    this.driverHealth = !this.driverHealth;
    this.isViolationCheckboxEmpty = false;
  }

  showViolations() {
    this.driverViolation = !this.driverViolation;
    this.isViolationCheckboxEmpty = false;
  }

  scrollToTarget(el: HTMLElement) {
    var bodyRect = document.body.getBoundingClientRect(),
      elemRect = el.getBoundingClientRect(),
      offset = elemRect.top - bodyRect.top - 300;

    window.scrollTo({
      top: offset,
      behavior: "smooth",
    });
  }

  validateCheckBoxes() {
    if (this.medicalConditions.length <= 0 && this.driverHealth) {
      this.additionalDriverInfoForm
        .get("medicalConditions")
        .setErrors({ incorrect: true });
      this.isMedicalCheckboxEmpty = true;
    } else {
      this.additionalDriverInfoForm
        .get("medicalConditions")
        .setValue(this.medicalConditions);

      this.isMedicalCheckboxEmpty = false;
    }
    if (this.violations.length <= 0 && this.driverViolation) {
      this.additionalDriverInfoForm
        .get("driverViolations")
        .setErrors({ incorrect: true });
      this.isViolationCheckboxEmpty = true;
    } else {
      this.additionalDriverInfoForm
        .get("driverViolations")
        .setValue(this.violations);
      this.isViolationCheckboxEmpty = false;
    }
  }

  watchIdentityNoToShowDate() {
    var currentMeladiDate = new Date().getFullYear();
    var currentHijriYear: any = new Intl.DateTimeFormat("ar-TN-u-ca-islamic", {
      year: "numeric",
    }).format(Date.now());

    this.additionalDriverInfoForm
      .get("identityNumber")
      .valueChanges.subscribe((identityNumber) => {
        if (identityNumber.toString().startsWith("2") && identityNumber.length > 9) {
          this.years = [];
          this.months = DropDownValues.MeladiMonths;
          var b = currentMeladiDate - 90;
         for (b; b <= currentMeladiDate - 17; b++) {
          this.years.push(b);
          }
          this.showDate = true;
        } else  if (identityNumber.toString().startsWith("1") && identityNumber.length > 9) {
          this.years = [];
          this.months = DropDownValues.higriMonths;
          var b = parseInt(currentHijriYear.substring(0, 4)) - 90;
        for (b; b <= parseInt(currentHijriYear.substring(0, 4)) - 17; b++) {
          this.years.push(b);
          }
          this.showDate = true;
    
        }
      });
  }

  validateIdentityNo() {
    this.f.identityNumber.valueChanges.subscribe((identityNumber) => {
      if (identityNumber.length == 10) {
        const currentDriversId =  this.insuranceSteps.userQuoteRequest.value.drivers;
        console.log(currentDriversId);
        for (var i = 0; i < currentDriversId.length; i++) {
          if (identityNumber == currentDriversId[i].identityNumber) {
            this.f.identityNumber.setErrors({ duplicate: true });
            break;
          } else {
            this.f.identityNumber.setErrors(null);
          }
        }
      }
    });
  }
  @HostListener("window:resize", ["$event"])
  getScreenSize(event?): void {
    // If browser window is resized below mid screen size width
    window.innerWidth <= 1000
      ? (this.isOnMidScreen = true)
      : (this.isOnMidScreen = false);
  }
