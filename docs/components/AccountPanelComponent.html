import { Component, OnInit } from '@angular/core';
import { UserService } from 'src/app/services/user/user.service';
import { Router } from '@angular/router';
import { AppRoutes } from 'src/app/models/app/AppRoutes';
import { ProfileSettingsTypes } from 'src/app/models/app/ProfileSettingsTypes';
import AppUtils from 'src/app/helpers/utils/AppUtils';
import { slideIn } from 'src/app/helpers/utils/animations';


@Component({
   selector: 'app-account-panel',
   templateUrl: './account-panel.component.html',
   styleUrls: ['./account-panel.component.css'],
   animations: [ slideIn ]
})


export class AccountPanelComponent implements OnInit {

   isSuccessMsgVisible: boolean;
   successSettingType: ProfileSettingsTypes;
   isLoading: boolean;


   constructor(
      private userService: UserService, private router: Router
   ) { }


   ngOnInit() {
      this.getUserProfileData();
   }


   getUserProfileData(): void {
      // Attempt to get the user profile data from local storage
      const storedProfileData = this.userService.getUserProfileData();

      if (!storedProfileData) {
         // Display loader
         this.isLoading = true;
         // Send a request to fetch it from server if not found
         this.userService.fetchUserProfileData().subscribe(
            profileData => {
               // Store it in local storage
               this.userService.setUserProfileData(profileData.data);
               // Hide loader
               this.isLoading = false;
            },
            error => {
               this.router.navigate([AppRoutes.error]);
               this.isLoading = false;
            }
         );
      }
   }


   showSuccessMsg(settingsType: ProfileSettingsTypes): void {
      console.log(settingsType);
      this.successSettingType = settingsType;
      this.isSuccessMsgVisible = true;
      AppUtils.delay(3000).then(() => this.isSuccessMsgVisible = false);
   }

}
