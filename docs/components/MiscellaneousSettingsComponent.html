import { Component, OnInit } from '@angular/core';
import { AuthService } from 'src/app/services/auth/auth.service';
import { FormGroup, Validators, FormControl } from '@angular/forms';
import { UserService } from 'src/app/services/user/user.service';
import { slideIn } from 'src/app/helpers/utils/animations';
import { Router, ActivatedRoute } from '@angular/router';
import { AppRoutes } from 'src/app/models/app/AppRoutes';
import { ApiResponse } from 'src/app/payload/responses/ApiResponse';
import { DefaultLangChangeRequest } from 'src/app/payload/requests/user/DefaultLangChangeReq';


@Component({
   selector: 'app-miscellaneous-settings',
   templateUrl: './miscellaneous-settings.component.html',
   styleUrls: ['./miscellaneous-settings.component.css'],
   animations: [ slideIn ]
})


export class MiscellaneousSettingsComponent implements OnInit {

   languageSettingsForm: FormGroup;

   userId: string;
   userDefaultLanguage: string;
   isUserLangAr: boolean;

   isLoading: boolean;
   isActive: boolean;


   constructor(
      private authService: AuthService, private userService: UserService,
      private activeRoute: ActivatedRoute, private router: Router
   ) { }


   ngOnInit() {
      this.initForm();
      this.getUserData();
      this.watchActiveSettingsPanel();
      this.watchSettingsRoute();
      this.watchLanguageChange();
   }


   private initForm(): void {
      this.languageSettingsForm = new FormGroup({
         defaultLanguage: new FormControl('ar', [Validators.required])
      });
   }


   private getUserData(): void {
      this.userDefaultLanguage = this.userService.getUserProfileData().defaultLang;
      switch (this.userDefaultLanguage) {
         case 'ar':
            this.isUserLangAr = true;
         break;
         case 'en':
            this.isUserLangAr = false;
         break;
         default:
            this.isUserLangAr = true;
         break;
      }
   }


   private watchActiveSettingsPanel(): void {
      this.userService.getActiveSettingsPanel().subscribe(panels => {
         if (panels.misc) {
            this.isActive = true;
         } else {
            this.isActive = false;
         }
      });
   }


   private watchSettingsRoute(): void {
      this.activeRoute.params.subscribe(param => {
         if (param.activePanel === 'misc') {
            this.isActive = true;
         }
      });
   }


   cancelEdit(): void {
      this.userService.resetActiveSettingsPanel();
      this.router.navigate([AppRoutes.controlPanel.accountSettings.uri]);
   }


   setActive(): void {
      // Set current panel as active
      this.userService.setActiveSettingsPanel({ password: false, email: false, phone: false, misc: true });
      this.router.navigate([AppRoutes.controlPanel.accountSettings.misc]);
   }


   watchLanguageChange(): void {
      this.languageSettingsForm.valueChanges.subscribe(selectedLanguage => {

         // Display loader & disable changing the selection
         this.isLoading = true;

         // Construct the phone change request
         const languageChangeReq: DefaultLangChangeRequest = {
            userId: this.authService.getTokenUserId(),
            language: selectedLanguage.defaultLanguage
         };

         // Send the request
         this.userService.changeDefaultLanguage(languageChangeReq).subscribe(
            (res: ApiResponse) => {
               if (res.isSuccess) {
                  // Stop loader
                  this.isLoading = false;
                  // Update the user's default language on the stored user profile data
                  const profileData = this.userService.getUserProfileData();
                  profileData.defaultLang = selectedLanguage.defaultLanguage;
                  this.userService.setUserProfileData(profileData);
                  // TODO: Display success message
               }
            },
            err => {
               const error: ApiResponse = err.error;
               // Enable the language select radios
               //this.languageSettingsForm.controls.defaultLanguage.enable();
               // Stop loader
               this.isLoading = false;
               // TODO: Display error message
            }
         );

      });
   }

}
