import {
  animate,
  state,
  style,
  transition,
  trigger,
} from "@angular/animations";
import { AfterViewInit, Component, HostListener, OnInit } from "@angular/core";
import { Subject } from "rxjs";
import { delay } from "rxjs/operators";
import { Driver } from "src/app/models/insurance-steps/Driver";
import { AuthService } from "src/app/services/auth/auth.service";
import { InsuranceStepsApiService } from "src/app/services/insurance-steps/insurance-steps-api.service";
import { InsuranceStepsService } from "src/app/services/insurance-steps/insurance-steps.service";
import { FormsValidationService } from "src/app/services/insurance-steps/forms-validation.service";
import { LocallyStoredItemsKeys } from 'src/app/models/app/LocallyStoredItemsKeys';

@Component({
  selector: "app-quotation-request-page",
  templateUrl: "./quotation-request-page.component.html",
  styleUrls: ["./quotation-request-page.component.css"],
  animations: [
    trigger("step", [
      state(
        "in",
        style({
          opacity: 1,
          transform: "translateX(0)",
        })
      ),
      transition("void => *", [
        style({
          opacity: 0,
          transform: "translateX(-100px)",
        }),
        animate(500),
      ]),
    ]),
  ],
})
export class QuotationRequestPageComponent implements OnInit, AfterViewInit {
  submitSubject: Subject<void> = new Subject<void>();
  stepNumber;
  showInsuranceInfo = true;
  showVehicleInfo = false;
  showDriverInfo = false;
  showadditionalDiver = false;
  showMoreDetails = false;
  showRequestBtn = false;
  currentDrivers;
  isLoadingQuotes = false;
  hasLoadingQuotesFailed = false;
  isOnMidScreen;
  driversPercentage = 0;
  userQuotationDrivers: Driver[] = [];
  inCompleteDrivingPercentage= false;
  percentageAlert = false;
  constructor(
    private insuranceStepService: InsuranceStepsService,
    private insuranceStepServiceApi: InsuranceStepsApiService,
    private authService: AuthService,
    private formvalidationService: FormsValidationService
  ) {}

  ngOnInit() {
    this.getScreenSize();
    this.insuranceStepServiceApi.getStepsData();
   
  }

  ngAfterViewInit() {
    this.watchStepNum();
  }

  watchStepNum() {
    this.insuranceStepService
      .getCurrentStepNum()
      .pipe(delay(100))
      .subscribe((stepNum) => {
        this.stepNumber = stepNum;
        if (stepNum >= 2) {
          this.showVehicleInfo = true;
        }
        if (stepNum >= 3) {
          this.showDriverInfo = true;
        }
        if (stepNum >= 4) {
          this.showMoreDetails = true;
        }
        if (stepNum >= 5) {
          this.showadditionalDiver = true;
        }
      });
  }

  delay(ms: number) {
    return new Promise((resolve) => setTimeout(resolve, ms));
  }

  sendQuotesRequest() {
    if (this.authService.getIsLoggedIn().value) {
      // to submit all steps forms to generate quotation request object
      /*  this.submitSubject.next(); */
      // Show the quotes list loader while waiting for request

      this.isLoadingQuotes = true;

      // Scroll to the top of the page
      window.scrollTo({ top: 0 });

      //  Fetch the quotes list
      this.insuranceStepServiceApi.getQuotationList().subscribe(
        (res: any) => {
          this.isLoadingQuotes = false;
          this.hasLoadingQuotesFailed = true;
          // remove quotation requet data from local storage
          localStorage.removeItem(
            LocallyStoredItemsKeys.InsuranceCurrentStepData,
          );
          localStorage.removeItem(
            LocallyStoredItemsKeys.UserQuoteRequest,
          );
          // Increment step number
          this.insuranceStepService.setCurrentStepNum(6);
          if (!res.isSuccess) {
            this.isLoadingQuotes = false;
            this.hasLoadingQuotesFailed = true;
          }
        },
        (err) => {
          this.isLoadingQuotes = false;
          this.hasLoadingQuotesFailed = true;
        }
      );
    } else {
      this.isLoadingQuotes = false;
      this.authService.setIsAuthModalShown(true);
      this.authService.getIsAuthModalShown().subscribe((isModalDisplayed) => {
        if (!isModalDisplayed) {
          this.sendQuotesRequest();
        }
      });
    }
  }

  async scrollToTarget(elId) {
    await this.delay(100);
    const el = document.getElementById(elId);
    if (el == null) {
      return;
    }
    var bodyRect = document.body.getBoundingClientRect(),
      elemRect = el.getBoundingClientRect(),
      offset = elemRect.top - bodyRect.top - 100;

    window.scrollTo({
      top: offset,
      behavior: "smooth",
    });
  }

  watchFormsBeforeSendRequest() {
    let formsStatus = this.formvalidationService.getFormsStatus().value;
    let invalidForm = formsStatus.filter((status) => status.isValid == false);
    // check the percentage of driver if equal 100%
      
    let percentagesSum = 0; // Indicates the sum of all additional drivers percentages in the array

    this.insuranceStepService.getUserQuoteRequest().value.drivers.forEach((driver) => {
      //  Calculate the sum of the  drivers
      percentagesSum += driver.drivingPercentageId;
    });

    if(percentagesSum < 100) {
      this.inCompleteDrivingPercentage = true;
      this.percentageAlert = false;
    }
    else  {
      if (invalidForm.length > 0) {
        this.scrollToTarget(invalidForm[0].formName);
      } else {
        this.sendQuotesRequest();
      }
    }
   
  }
  

  @HostListener("window:resize", ["$event"])
  getScreenSize(event?): void {
    // If browser window is resized below mid screen size width
    window.innerWidth >= 1000
      ? (this.isOnMidScreen = true)
      : (this.isOnMidScreen = false);
  }
}
