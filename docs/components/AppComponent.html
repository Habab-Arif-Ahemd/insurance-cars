import { Component, OnInit } from '@angular/core';
import { InsuranceStepsService } from './services/insurance-steps/insurance-steps.service';
import { AuthService } from './services/auth/auth.service';
import { TranslateService, LangChangeEvent } from '@ngx-translate/core';


@Component({
   selector: 'app-root',
   templateUrl: './app.component.html',
   styleUrls: ['./app.component.css']
})


export class AppComponent implements OnInit {


   isAuthModalVisible = false;
   completionPercentage = '0%';
   locale: string;


   constructor(
      private insuranceStepService: InsuranceStepsService,
      private authService: AuthService,
      private translateService: TranslateService
   ) { }


   ngOnInit() {
      this.watchInsuranceStepsProgress();
      this.watchLoginModalState();
      this.locale = this.translateService.currentLang;
      this.translateService.onLangChange
      .subscribe((langChangeEvent: LangChangeEvent) => {
         this.locale = langChangeEvent.lang;
      });
   }


   watchInsuranceStepsProgress(): void {
      this.insuranceStepService.getCurrentStepNum().subscribe(stepNum => {
         if (stepNum <= this.insuranceStepService.totalInsuranceSteps) {
            this.completionPercentage = Math.floor(((stepNum / this.insuranceStepService.totalInsuranceSteps) * 100)).toString() + '%';
         }
      });
   }


   watchLoginModalState(): void {
      this.authService.getIsAuthModalShown().subscribe(isAuthModalVisible => this.isAuthModalVisible = isAuthModalVisible)
   }

}
