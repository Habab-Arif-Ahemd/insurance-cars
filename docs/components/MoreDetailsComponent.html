import {
  ChangeDetectionStrategy,
  Component,
  EventEmitter,
  HostListener,
  OnDestroy,
  OnInit,
  Output,
} from "@angular/core";
import { FormBuilder, FormGroup, Validators } from "@angular/forms";
import { Scroll } from "src/app/helpers/utils/scroll";
import { Driver } from "src/app/models/insurance-steps/Driver";
import { DriverLicense } from "src/app/models/insurance-steps/DriverLicense";
import { DriverViolation } from "src/app/models/insurance-steps/DriverViolation";
import { MedicalCondition } from "src/app/models/insurance-steps/MedicalCondition";
import { QuoteRequest } from "src/app/models/insurance-steps/QuoteRequest";
import { AppService } from "src/app/services/app.service";
import { InsuranceStepsService } from "src/app/services/insurance-steps/insurance-steps.service";

@Component({
  changeDetection: ChangeDetectionStrategy.Default,
  selector: "app-more-details",
  templateUrl: "./more-details.component.html",
  styleUrls: [
    "./more-details.component.css",
    "../quotation-request-page/quotation-request-page.component.css",
  ],
})
export class MoreDetailsComponent implements OnInit {
 
  @Output() scroll = new EventEmitter();
  moreDriverInfoForm: FormGroup;
  driverLicenseForm: FormGroup;
  driverLicense = false;
  driverViolation = false;
  driverHealth = false;
  addedlicense = false;
  driverVehicleSpecification = false;

  addingDriver = false;

  hasChildren = false;
  stepNumber: any;
  stepsData;

  driverLicenseFormSubmitted: boolean;
  licenseslist = [];
  foreignlicenses = [];
  medicalConditions: any = [];
  violations = [];
  vehicleSpecification: any = [];
  noOfYears = [
    { id: 1, name: "1" },
    { id: 2, name: "2" },
    { id: 3, name: "3" },
    { id: 4, name: "4" },
    { id: 5, name: "5" },
    { id: 6, name: "6" },
    { id: 7, name: "7" },
    { id: 8, name: "8" },
    { id: 9, name: "9" },
    { id: 10, name: "10" },
  ];

  isMedicalCheckboxEmpty: boolean;
  isViolationCheckboxEmpty: boolean;
  isVehicleSpecificationEmpty: boolean;

  lang: string;
  isOnMidScreen: boolean;

  previousQuote: QuoteRequest;
  previousQuoteDriver: Driver;
  previousQuoteDriverMedicalId;
  previousQuoteDriverViolationsId;
  previousQuoteDriverLincensesId;
  isSaudiLicense: any;

  constructor(
    private insuranceSteps: InsuranceStepsService,
    private fb: FormBuilder,
    private scrollOnErrer: Scroll,
    private appService: AppService
  ) {}

  ngOnInit() {
    this.initForm();
    this.appService.getAppLang().subscribe((language) => {
      this.lang = language;
    });
    this.watchStepNum();
  }

  initForm() {
    this.moreDriverInfoForm = this.fb.group({
      medicalConditions: [null],
      driverLicenses: [null],
      driverViolations: [null],
      vehicleSpecification: [null],
    });
    this.initdriverLicensesForm();
    this.getStepData();
  }

  initdriverLicensesForm() {
    this.driverLicenseForm = this.fb.group({
      licenseTypeId: [null],
      countryId: [null],
      numberOfYears: [null],
    });
  }
  get df() {
    return this.driverLicenseForm.controls;
  }

  submitForm(): void {
    this.validateCheckBoxes();
    if (this.moreDriverInfoForm.invalid) {
      const firstElementWithError = document.querySelector(".v-meassage");
      this.scrollOnErrer.scrollTo(firstElementWithError);
      return;
    }

    this.moreDriverInfoForm.value.medicalConditions = this.medicalConditions;
    this.moreDriverInfoForm.value.driverLicenses = this.foreignlicenses;
    this.moreDriverInfoForm.value.driverViolations = this.violations;
    this.moreDriverInfoForm.value.vehicleSpecification = this.vehicleSpecification;
    /* check if toggle is off remove medical  and violation  */

    if (!this.driverHealth) {
      this.moreDriverInfoForm.value.medicalConditions = [];
    }
    if (!this.driverViolation) {
      this.moreDriverInfoForm.value.driverViolations = [];
    }

    const tempQuoteReq = this.insuranceSteps.getUserQuoteRequest().value;
    tempQuoteReq.vehicleSpecifications = this.moreDriverInfoForm.value.vehicleSpecification;
    const tempDrivers = tempQuoteReq.drivers.concat();
    tempDrivers.forEach((driver) => {
      if (driver.typeId === 1) {
        driver.medicalConditions = this.moreDriverInfoForm.value.medicalConditions;
        driver.driverViolations = this.moreDriverInfoForm.value.driverViolations;
        driver.driverLicenses = this.moreDriverInfoForm.value.driverLicenses;
      }
    });

    this.insuranceSteps.overridenSetQuoteRequest(tempQuoteReq);
    this.stepNumber = this.insuranceSteps.setCurrentStepNum(5);
  }

  removeForm(licenseData: any): void {
    const index: number = this.foreignlicenses.indexOf(licenseData);
    if (index !== -1) {
      this.foreignlicenses.splice(index, 1);
    }
  }

  getStepData() {
    this.insuranceSteps.getInsuranceStepsData().subscribe((stepsData) => {
      this.stepsData = stepsData;

      if (
        this.stepsData.MedicalConditions &&
        this.stepsData.MedicalConditions.length == 9 &&
        this.stepsData.violations &&
        this.stepsData.violations.length >= 4
      ) {
        this.getPreviousQuoteRequest();
      }
    });
  }

  addLicense(id: DriverLicense) {
    
    id.isSaudiLicense = false;
    this.driverLicenseFormSubmitted = true;
    for (let i = 0; i < this.foreignlicenses.length; i++) {
      if (
        this.foreignlicenses[i].licenseTypeId === id.licenseTypeId &&
        this.foreignlicenses[i].countryId === id.countryId
      ) {
        this.df.licenseTypeId.setErrors({'duplicate': true});
        break;
      } 
    }
   

    if (this.driverLicenseForm.invalid) {
      return;

    } else {
        const country = this.stepsData.Countries.filter(
            (country) => country.id === id.countryId
          )[0];
          const licenseType = this.stepsData.LicenseTypes.filter(
            (type) => type.id === id.licenseTypeId
          )[0];
          const noOfYear = this.noOfYears.filter(
            (no) => no.id === id.numberOfYears
          )[0];
          const driverLicense = {
            country: country,
            licenseType: licenseType,
            noOfYear: noOfYear,
            isSaudiLicense: false,
          };
          this.licenseslist.push(driverLicense);
          this.foreignlicenses.push(id);
          this.driverLicenseForm.reset();
          this.driverLicenseFormSubmitted = false;
    }
   
  }

  removeFromLicenseList(licenseData: any): void {
    // Remove from licenses list
    const index: number = this.licenseslist.indexOf(licenseData);
    if (index !== -1) {
      this.licenseslist.splice(index, 1);
    }

    // Construct foreign license object to be removed from the array
    const licenseToRemove = {
      licenseTypeId: licenseData.licenseType.id,
      countryId: licenseData.country.id,
      numberOfYears: licenseData.noOfYear.id,
      isSaudiLicense: licenseData.isSaudiLicense,
    };
    // Find the index of the foreign license to be removed in the foreign licenses array
    let foreignLicenseIdx = -1;
    for (let i = 0; i < this.foreignlicenses.length; i++) {
      if (
        JSON.stringify(this.foreignlicenses[i]) ===
        JSON.stringify(licenseToRemove)
      ) {
        foreignLicenseIdx = i;
      }
    }

    // Remove the license from the foreign licenses array
    if (foreignLicenseIdx !== -1) {
      this.foreignlicenses.splice(foreignLicenseIdx, 1);
    }
  }

  addViolation(id: number) {
    this.isViolationCheckboxEmpty = false;
    var index = this.violations
      .map((e) => {
        return e.violationId;
      })
      .indexOf(id);
    if (index === -1) {
      this.violations.push({ violationId: id });
    } else {
      this.violations.splice(index, 1);
    }
  }

  addMedicalCondition(id: number) {
    this.isMedicalCheckboxEmpty = false;
    this.moreDriverInfoForm.get("medicalConditions").setErrors(null);
    var index = this.medicalConditions
      .map((e) => {
        return e.medicalConditionId;
      })
      .indexOf(id);

    if (index === -1) {
      this.medicalConditions.push({ medicalConditionId: id });
    } else {
      this.medicalConditions.splice(index, 1);
    }
  }

  addVehicleSpecification(id: number) {
    this.isVehicleSpecificationEmpty = false;
    this.moreDriverInfoForm.get("vehicleSpecification").setErrors(null);
    var index = this.vehicleSpecification
      .map((e) => {
        return e.vehicleSpecificationId;
      })
      .indexOf(id);

    if (index === -1) {
      this.vehicleSpecification.push({ vehicleSpecificationId: id });
    } else {
      this.vehicleSpecification.splice(index, 1);
    }
  }

  showMedicalConditions() {
    this.driverHealth = !this.driverHealth;
    this.isMedicalCheckboxEmpty = false;
  }

  showViolations() {
    this.driverViolation = !this.driverViolation;
    this.isViolationCheckboxEmpty = false;
  }

  showVehicleSpecifications() {
    this.driverVehicleSpecification = !this.driverVehicleSpecification;
    this.isVehicleSpecificationEmpty = false;
  }

  showLicenseForm() {
    this.initdriverLicensesForm();
    this.driverLicense = !this.driverLicense;
    this.driverLicenseFormSubmitted = false;
    this.driverLicenseForm.reset();
    this.foreignlicenses.length = 0;
    this.licenseslist.length = 0;

    if (this.driverLicense) {
      this.driverLicenseForm.controls["licenseTypeId"].setValidators([
        Validators.required,
      ]);

      this.driverLicenseForm.controls["countryId"].setValidators([
        Validators.required,
      ]);
      this.driverLicenseForm.controls["numberOfYears"].setValidators([
        Validators.required,
      ]);
      this.driverLicenseForm.updateValueAndValidity();
    } else {
      this.driverLicenseForm.controls["licenseTypeId"].clearValidators();
      this.driverLicenseForm.controls["countryId"].clearValidators();
      this.driverLicenseForm.controls["numberOfYears"].clearValidators();
      this.driverLicenseForm.updateValueAndValidity();
    }
  }

  watchStepNum() {
    this.insuranceSteps.getCurrentStepNum().subscribe((stepNum) => {
      this.stepNumber = stepNum;
    });
  }

  validateCheckBoxes() {
    if (
      this.medicalConditions.length <= 0 &&
      this.driverHealth &&
      this.previousQuoteDriver.medicalConditions.length <= 0
    ) {
      this.moreDriverInfoForm
        .get("medicalConditions")
        .setErrors({ incorrect: true });
      this.isMedicalCheckboxEmpty = true;
    } else {
      this.moreDriverInfoForm
        .get("medicalConditions")
        .setValue(this.medicalConditions);

      this.isMedicalCheckboxEmpty = false;
    }
    if (
      this.violations.length <= 0 &&
      this.driverViolation &&
      this.previousQuoteDriver.driverViolations.length <= 0
    ) {
      this.moreDriverInfoForm
        .get("driverViolations")
        .setErrors({ incorrect: true });
      this.isViolationCheckboxEmpty = true;
    } else {
      this.moreDriverInfoForm.get("driverViolations").setValue(this.violations);
      this.isViolationCheckboxEmpty = false;
    }
  }

 

  getPreviousQuoteRequest() {
    let previousQuoteDriverMedical: MedicalCondition[];
    let previousQuoteDriverViolations: DriverViolation[];
    let previousQuoteDriverLicenses: DriverLicense[];
    this.previousQuote = this.insuranceSteps.getUserQuoteRequest().value;

    if (this.previousQuote.drivers) {
      this.previousQuote.drivers.forEach((drivers) => {
        if (drivers.typeId == 1) {
          this.previousQuoteDriver = drivers;
        }
      });
      //  Fetch PreviousQuotation Driver medicalConditions ids
      previousQuoteDriverMedical = this.previousQuoteDriver.medicalConditions;
      this.previousQuoteDriver.medicalConditions.length > 0
        ? ((this.driverHealth = true),
          (previousQuoteDriverMedical = this.previousQuoteDriver
            .medicalConditions))
        : (this.driverHealth = false);

      previousQuoteDriverMedical.forEach((medicalCondition) => {
        this.stepsData.MedicalConditions.forEach((condition) => {
          if (medicalCondition.medicalConditionId === condition.id) {
            condition.checked = true;
          }
        });
        this.previousQuoteDriverMedicalId = medicalCondition.medicalConditionId;
      });

      //  Fetch PreviousQuotation Driver Violation ids

      if (this.previousQuoteDriver.driverViolations.length > 0) {
        this.driverViolation = true;
        previousQuoteDriverViolations = this.previousQuoteDriver
          .driverViolations;
        previousQuoteDriverViolations.forEach((driverViolations) => {
          this.stepsData.violations.forEach((violation) => {
            if (driverViolations.violationId == violation.id) {
              violation.checked = true;
            }
          });
          this.previousQuoteDriverViolationsId = driverViolations.violationId;
        });
      } else {
        this.driverViolation = false;
      }

      //  Fetch PreviousQuotation Driver Licenses ids
      if (this.previousQuoteDriver.driverLicenses.length > 1) {
        this.showLicenseForm();
        this.driverLicense = true;
        this.previousQuoteDriverLincensesId = true;
        previousQuoteDriverLicenses = this.previousQuoteDriver.driverLicenses;
        previousQuoteDriverLicenses.forEach((license) => { 
          if (!license.isSaudiLicense) {
            this.addLicense(license);
          }
        });
      } else {
        this.driverLicense = false;
      }
    }
  }


  @HostListener("window:resize", ["$event"])
  getScreenSize(event?): void {
    // If browser window is resized below mid screen size width
    window.innerWidth >= 1000
      ? (this.isOnMidScreen = true)
      : (this.isOnMidScreen = false);
  }
}
