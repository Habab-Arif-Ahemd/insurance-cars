import { Component, OnInit } from "@angular/core";
import { ProfileService } from "src/app/services/profile/profile.service";
import { ProfileSavedQuote } from "src/app/models/quote/SavedQuote";
import { PreviewQuoteResponse } from 'src/app/payload/responses/quotes/PreviewQuoteResponse';
import { QuoteService } from 'src/app/services/quote/quote.service';
import { LocallyStoredItemsKeys } from 'src/app/models/app/LocallyStoredItemsKeys';
import { Router } from '@angular/router';
import { ApiRoutes } from 'src/app/models/app/ApiRoutes';
import { AppRoutes } from 'src/app/models/app/AppRoutes';

@Component({
  selector: "app-prev-quotes",
  templateUrl: "./prev-quotes.component.html",
  styleUrls: ["./prev-quotes.component.css"],
})
export class PrevQuotesComponent implements OnInit {
  isLoadingUserPolicies = false;
  savedQuotes: ProfileSavedQuote[];

  constructor(private profileService: ProfileService, private router: Router) {}

  ngOnInit() {
    window.scrollTo(0, 0);
    this.getSavedQuote();
  }

  getSavedQuote() {
    this.isLoadingUserPolicies = true;
    this.profileService.getSavedQuotes().subscribe((savedQuote: ProfileSavedQuote) => {
      this.savedQuotes = savedQuote.data;
      savedQuote.btnLoader = false;
      this.isLoadingUserPolicies= false;
    });
  }

  buySavedQuote(id: string) {
    this.savedQuotes.forEach(qoute => {
      qoute.btnLoader = false;
      if(id == qoute.id) {
        qoute.btnLoader = true;
      }
    })
    this.profileService.getSavedQuote(id).subscribe((res: PreviewQuoteResponse) => {
      localStorage.setItem(LocallyStoredItemsKeys.PreviewQuoteResponse, JSON.stringify(res));
      this.router.navigate([AppRoutes.checkout]);
    })
  }

}
