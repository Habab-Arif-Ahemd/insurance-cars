import { Component, OnInit } from "@angular/core";
import { FormBuilder, FormGroup, Validators } from "@angular/forms";
import { Router } from "@angular/router";
import { NationalIdValidator } from "src/app/helpers/validators/NationalIdValidator";
import { ProfileService } from 'src/app/services/profile/profile.service';
import { VehicleInfo } from 'src/app/models/VehicleInfo';
import { InquireService } from 'src/app/services/insurance-steps/inquire.service';
import { PreliminaryDataRequest } from 'src/app/payload/requests/quotation/PreliminaryDataRequest';

@Component({
  selector: "app-vehicles",
  templateUrl: "./vehicles.component.html",
  styleUrls: ["./vehicles.component.css"]
})
export class VehiclesComponent implements OnInit {
  vehicleInfo: VehicleInfo;
  submitted = false;
  vehicleFormGroup: FormGroup;

  showVehicleList = true;
  showAddingVehicleForm = false;
  isLoading= false;
  errorMessage: any;
  isLoadingVehicleList = false;
  constructor(
    private fb: FormBuilder,
    private router: Router,
    private profileService: ProfileService,
    private inquireService: InquireService
  ) {}

  ngOnInit() {
    this.getVehicles();
    this.initForm();
  }

  initForm() {
    this.vehicleFormGroup = this.fb.group({
      identityNumber: [
        "",
        Validators.compose([
          Validators.required,
          NationalIdValidator,
          Validators.minLength(10)
        ])
      ],
      vehicleId: ["", Validators.required],
      vehicleIdTypeId: [1, [Validators.required]]
    });
  }

  onSubmit() {
    this.submitted = true;
    // Show loading spinner
    this.isLoading = true;

    // Validate form
    if (this.vehicleFormGroup.invalid) {
      this.isLoading = false;
      return;
    }
    // Construct request data which is used to retreive the preliminary quote order data
    this.inquireService.inquire(this.vehicleFormGroup.value);
  }

 inquire(insuredIdentityNumber, vehicleId) {
   this.isLoading= true;
   let inquireRequest: PreliminaryDataRequest = {insuredIdentityNumber, vehicleId};
   this.inquireService.inquire(inquireRequest);
  // this.isLoading= false;


 }
  // convenience getter for easy access to form fields
  get f() {
    return this.vehicleFormGroup.controls;
  }

  getVehicles() {
    this.isLoadingVehicleList = true;
     this.profileService.getVehiclesList().subscribe(vehicleInfo => {
        this.vehicleInfo= vehicleInfo;
        this.isLoadingVehicleList = false;
     })
   
  }

  showVehiclesList() {
    this.submitted = false;
    this.showAddingVehicleForm = false;
    this.showVehicleList = true;
  }

  showAddingVehiclesForm() {
    this.showAddingVehicleForm = true;
    this.showVehicleList = false;
  }

  submitNewVehicle() {
    this.submitted = true;
  }
}
