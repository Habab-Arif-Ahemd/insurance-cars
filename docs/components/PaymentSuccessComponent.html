import { Component, OnInit } from '@angular/core';
import { OrderReviewService } from 'src/app/services/order-review/order-review.service';
import { ActivatedRoute } from '@angular/router';


@Component({
   selector: 'app-payment-success',
   templateUrl: './payment-success.component.html',
   styleUrls: ['./payment-success.component.css']
})


export class PaymentSuccessComponent implements OnInit {

   constructor(
      private orderReviewService: OrderReviewService,
      private activatedRoute: ActivatedRoute
   ) {
      this.activatedRoute.queryParams.subscribe(params => {
         this.getCheckoutStatus(params['id']);
      });
   }


   ngOnInit() { }


   private getCheckoutStatus(checkoutId: string) {
      this.orderReviewService.getCheckoutStatus(checkoutId).subscribe(
         res => console.log(res),
         err => console.log(err)
      );
   }

}
