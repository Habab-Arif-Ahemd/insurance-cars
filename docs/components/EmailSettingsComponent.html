import { Component, OnInit, EventEmitter, Output } from '@angular/core';
import { AppService } from 'src/app/services/app.service';
import { AuthService } from 'src/app/services/auth/auth.service';
import { FormGroup, Validators, FormControl } from '@angular/forms';
import { UserService } from 'src/app/services/user/user.service';
import { slideIn } from 'src/app/helpers/utils/animations';
import { Router, ActivatedRoute } from '@angular/router';
import { Location } from '@angular/common';
import { AppRoutes } from 'src/app/models/app/AppRoutes';
import { UsernameChangeRequest } from 'src/app/payload/requests/user/UsernameChangeReq';
import { ApiResponse } from 'src/app/payload/responses/ApiResponse';
import { UsernameChangeVerificationReq } from 'src/app/payload/requests/user/UsernameChangeVerificationReq';
import { ProfileSettingsTypes } from 'src/app/models/app/ProfileSettingsTypes';


@Component({
   selector: 'app-email-settings',
   templateUrl: './email-settings.component.html',
   styleUrls: ['./email-settings.component.css'],
   animations: [ slideIn ]
})


export class EmailSettingsComponent implements OnInit {

   /* Events */
   @Output() successEmitter = new EventEmitter<ProfileSettingsTypes>();

   /* Forms */
   emailSettingsForm: FormGroup;

   /* Data */
   userEmail: any;

   /* UI */
   isLoading: boolean;
   isActive: boolean;
   isVerifying: boolean;
   isSubmitting: boolean;

   /* Error */
   isErrorAlertVisible: boolean;
   errorMessage: string;


   constructor(
      private authService: AuthService, private userService: UserService,
      private activeRoute: ActivatedRoute, private router: Router,
      private location: Location
   ) { }


   ngOnInit() {
      this.initForm();
      this.getUserData();
      this.watchActiveSettingsPanel();
      this.watchSettingsRoute();
   }


   /* INITIALIZATION */

   private initForm(): void {
      this.emailSettingsForm = new FormGroup({
         userName: new FormControl(null, [Validators.required, Validators.email]),
         password: new FormControl(null, [Validators.required, Validators.minLength(8)]),
         verificationCode: new FormControl(null)
      });
   }

   get emailForm() {
      return this.emailSettingsForm.controls;
   }

   private getUserData(): void {
      this.userEmail = this.userService.getUserProfileData().email;
   }



   /* UI & ROUTING */

   private watchActiveSettingsPanel(): void {
      this.userService.getActiveSettingsPanel().subscribe(panels => {
         if (panels.email) {
            this.isActive = true;
         } else {
            this.isActive = false;
         }
      });
   }

   private watchSettingsRoute(): void {
      this.activeRoute.params.subscribe(param => {
         if (param.activePanel === 'email') {
            this.isActive = true;
            this.isVerifying = false;
         }
         if (param.state === 'verify') {
            this.isVerifying = true;
         }
      });
   }

   cancelEdit(): void {
      this.userService.resetActiveSettingsPanel();
      this.router.navigate([AppRoutes.controlPanel.accountSettings.uri]);
   }

   setActive(): void {
      // Set current panel as active
      this.userService.setActiveSettingsPanel({ password: false, email: true, phone: false, misc: false });
      this.router.navigate([AppRoutes.controlPanel.accountSettings.email.main]);
   }



   /* FORMS SUBMIT */
   
   submitEmailChange(): void {

      // Display loader
      this.isLoading = true;
      this.isSubmitting = true;

      // Validate inputs
      if (this.emailSettingsForm.invalid) {
         this.isLoading = false;
         return;
      }

      // Construct the email (email = username) change request
      const emailChangeReq: UsernameChangeRequest = {
         userId: this.authService.getTokenUserId(),
         userName: this.emailForm.userName.value,
         password: this.emailForm.password.value
      };

      // Send the request
      this.userService.requestUsernameChange(emailChangeReq).subscribe(
         (res: ApiResponse) => {
            if (res.isSuccess) {

               // Stop loader/submitting
               this.isLoading = false;
               this.isSubmitting = false;

               // Change URL to '/verify' without navigating to keep UI state on refresh
               this.location.go(AppRoutes.controlPanel.accountSettings.email.verify);

               // Display the verification code input
               this.isVerifying = true;

            }
         },
         error => this.showErrorAlert(error)
      );

   }

   submitEmailChangeVerification(): void {

      // Display loader
      this.isLoading = true;
      this.isSubmitting = true;

      // Set the password control value so it validates (since it could be reset on refresh. It will be hidden, and its value will not be used)
      this.emailForm.password.setValue('0');

      // Validate inputs verification code
      if (!this.emailForm.userName.value || !this.emailForm.verificationCode.value) {
         this.isLoading = false;
         return;
      }

      // Construct the change verification request
      const emailChangeVerificationReq: UsernameChangeVerificationReq = {
         userId: this.authService.getTokenUserId(),
         userName: this.emailForm.userName.value,
         verificationCode: this.emailForm.verificationCode.value
      };

      // Send the request
      this.userService.changeUsername(emailChangeVerificationReq).subscribe(
         (res: ApiResponse) => {
            if (res.isSuccess) {

               // Stop loader/submitting
               this.isLoading = false;
               this.isSubmitting = false;

               // Hide the verification code input
               this.isVerifying = false;

               // Update the user's email on the stored user profile data
               const profileData = this.userService.getUserProfileData();
               profileData.email = this.emailForm.userName.value;
               this.userService.setUserProfileData(profileData);

               // Show success message/art
               this.isActive = false;
               this.successEmitter.emit(ProfileSettingsTypes.EmailSettings);
               
            }
         },
         error => this.showErrorAlert(error)
      );

   }



   /* MISC */

   private showErrorAlert(errorResponse: any): void {
      // Get error data
      const error: ApiResponse = errorResponse.error;

      // Stop the loader
      this.isLoading = false;
      this.isSubmitting = false;

      // Get error message
      if (error.validationErrors && error.validationErrors.length > 0) {
         this.errorMessage = error.validationErrors[0].description;
      } else if (!error.validationErrors && error.responseMessage) {
         this.errorMessage = error.responseMessage;
      } else {
         this.errorMessage = 'An error has occurred';
      }

      // Show error message
      if (this.errorMessage.length > 0) {
         this.isErrorAlertVisible = true;
      }
   }

}
